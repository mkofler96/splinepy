38 template<std::size_t para_dim, std::size_t dim>
39 typename Bezier<para_dim, dim>::Base_
40 Bezier<para_dim, dim>::CreateBase(const int* degrees,
41  const double* control_points) {
42 
43  std::array<std::size_t, para_dim> bm_degrees{};
44  std::size_t ncps{1};
45 
46  // formulate degrees
47  for (std::size_t i{}; i < para_dim; ++i) {
48  bm_degrees[i] = degrees[i];
49  ncps *= degrees[i] + 1;
50  }
51 
52  // formulate control_points
53  std::vector<Coordinate_> bm_control_points(ncps);
54  for (std::size_t i{}; i < ncps; ++i) {
55  if constexpr (dim > 1) {
56  for (std::size_t j = 0; j < dim; j++) {
57  bm_control_points[i][j] = control_points[i * dim + j];
58  }
59  } else {
60  bm_control_points[i] = control_points[i];
61  }
62  }
63  return Base_(bm_degrees, bm_control_points);
64 }

284  const std::shared_ptr<SplinepyBase>& a) const {
285 
286  SplinepySplineNameMatches(

577 template<std::size_t para_dim, std::size_t dim>
578 std::vector<std::shared_ptr<SplinepyBase>>
579 Bezier<para_dim, dim>::SplinepySplit(const int& p_dim,
580  const double& location) const {
581  // split
582  auto bm_splitted =
583  Base_::SplitAtPosition(location, static_cast<std::size_t>(p_dim));
584 
585  // make it splinepybase
586  std::vector<std::shared_ptr<SplinepyBase>> split;
587  const std::size_t n_splitted = bm_splitted.size();

658  this->SplinepyDerivativeSpline(order_query.data())

661
