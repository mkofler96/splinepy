45 template<typename SplineType, typename VectorType>
46 std::shared_ptr<splinepy::splines::SplinepyBase>
47 ExtractControlMeshSliceFromIDs(const SplineType& spline,
48  const VectorType& indices,
49  const int& plane_normal_axis) {
50  // Perform sanity check
51  if constexpr (SplineType::kParaDim == 1) {
52  splinepy::utils::PrintWarning(
53  "Sorry, we don't support control mesh slicing"
54  "of 1-Parametric Dim splines. Returning empty spline.");
55  return std::make_shared<SplineType>();
56  } else {
57  // boundary spline type
58  using SelfBoundary = typename SplineType::BoundaryType_;
59 
60  // prepare return
61  std::shared_ptr<SelfBoundary> boundary_spline;
62 
63  // create new one
64  if constexpr (SplineType::kHasKnotVectors) {
65  // hola BSpline families
66  using VSpace = typename SelfBoundary::PhysicalSpace_;
67 
68  // slices splines can use parent's parameter space without
69  // recreating basis.
70  // RemoveOneParametricDimension() creates a new parameter space with
71  // same shared pointers to all the required properties.
72  auto pspace = spline.GetParameterSpace().RemoveOneParametricDimension(
73  plane_normal_axis);
74 
75  // get size - dim should come from the coordinate itself, since nurbs have
76  // dim + 1 shape
77  const auto& from_coordinates = spline.GetCoordinates();
78  const int dim = from_coordinates.Shape()[1];
79  const int n_cps = indices.size();
80 
81  // VSpace - maybe it is also
82  // create (weighted) vector space and allocate space for cps
83  auto vspace = std::make_shared<VSpace>();
84  auto& coords = vspace->GetCoordinates();
85  coords.Reallocate(n_cps * dim);
86  coords.SetShape(n_cps, dim);
87 
88  // copy loop - copy each rows
89  int local_counter{};
90  for (const auto& id : indices) {
91  std::copy_n(&from_coordinates(id, 0), dim, &coords(local_counter++, 0));
92  }
93 
94  // assign boundary spline - uses base' ctor
95  boundary_spline = std::make_shared<SelfBoundary>(pspace, vspace);
96  } else {
97  // hola bezier families
98  // form degrees
99  using Degrees = typename SelfBoundary::Degrees_;
100  Degrees b_degrees{};
101  std::size_t ncps{1};
102  int counter{};
103  for (int i{}; i < SplineType::kParaDim; ++i) {
104  if (i == plane_normal_axis) {
105  continue;
106  }
107  const auto& ds = spline.GetDegrees();
108  const auto& d = ds[i];
109  b_degrees[counter] = d;
110  ncps *= static_cast<std::size_t>(d + 1);
111  counter++;
112  }
113 
114  if constexpr (SplineType::kIsRational) {
115  // only way to avoid unweight -> reweight is to init first with degrees
116  // assign boundary spline
117  boundary_spline = std::make_shared<SelfBoundary>(b_degrees);
118  auto& b_cps = boundary_spline->GetWeightedControlPoints();
119  auto& b_ws = boundary_spline->GetWeights();
120 
121  // copy
122  for (std::size_t i{}; i < ncps; ++i) {
123  const auto& id = indices[i];
124  b_cps[i] = spline.GetWeightedControlPoints()[id];
125  b_ws[i] = spline.GetWeights()[id];
126  }
127  } else {
128  // non-rational bezier
129  using Coordinates = typename SelfBoundary::Coordinates_;
130  Coordinates b_coords;
131  b_coords.reserve(ncps);
132  for (const auto& id : indices) {
133  // copy control points
134  b_coords.push_back(spline.GetControlPoints()[id]);
135  }
136 
137  // assign boundary spline
138  boundary_spline = std::make_shared<SelfBoundary>(b_degrees, b_coords);
139  }
140  }
141  // return here, this way compiler is happy
142  return boundary_spline;
143  }
144 }

146 template<typename SplineType>
147 std::shared_ptr<splinepy::splines::SplinepyBase>
148 ExtractBoundaryMeshSlice(const SplineType& spline, const int& boundary_id) {
149  if constexpr (SplineType::kParaDim == 1) {
150  splinepy::utils::PrintWarning(
151  "Sorry, we don't support control mesh slicing"
152  "of 1-Parametric Dim splines. Returning empty spline.");
153  return std::make_shared<SplineType>();
154  } else {
155 
156  // get ids on boundary
157  const auto cmr =
158  splinepy::splines::helpers::template GetControlMeshResolutions<int>(
159  spline);
160  const int plane_normal_axis = static_cast<int>(boundary_id / 2);
161  const int plane_id =
162  ((boundary_id % 2) == 0) ? 0 : cmr[plane_normal_axis] - 1;
163  const auto ids_on_boundary = splinepy::utils::GridPoints::IdsOnHyperPlane(
164  cmr.data(),
165  static_cast<int>(cmr.size()),
166  plane_normal_axis,
167  plane_id);
168  return ExtractControlMeshSliceFromIDs(spline,
169  ids_on_boundary,
170  plane_normal_axis);
171  }

225 template<typename SplineType>
226 std::vector<std::shared_ptr<splinepy::splines::SplinepyBase>>
227 ExtractBezierPatches(const SplineType& spline) {
228  static_assert(SplineType::kHasKnotVectors,
229  "Invalid type for ExtractBezierPatches");
230  // make copy of input spline
231  auto input_ptr = spline.SplinepyDeepCopy();
232  SplineType& input = *(std::dynamic_pointer_cast<SplineType>(input_ptr));
233 
234  // Start by identifying types
235  constexpr int para_dim = SplineType::kParaDim;
236  constexpr bool is_rational = SplineType::kIsRational;
237  const int dim = input.SplinepyDim();
238 
239  // Predetermine some auxiliary values
240  // Access spline degrees and determine offsets
241  std::array<int, para_dim> degrees{};
242  input.SplinepyCurrentProperties(degrees.data(), nullptr, nullptr, nullptr);
243 
244  std::array<int, para_dim> n_ctps_per_para_dim{};
245 
246  // Identify all internal knots and the number of required Bezier patches
247  const auto& parameter_space = input.GetParameterSpace();
248  for (int i_p_dim{}; i_p_dim < para_dim; i_p_dim++) {
249  // Need to use BSplineLib's infamous NamedType
250  const bsplinelib::Dimension pd_query{i_p_dim};
251 
252  // Extract internal knots
253  const auto bezier_information =
254  parameter_space.DetermineBezierExtractionKnots(pd_query);
255  const auto& required_knot_insertions = std::get<1>(bezier_information);
256 
257  // Insert knot into the copy of the spline before extraction
258  // this is the most costly part of the calculation
259  for (const auto& knot_to_insert : required_knot_insertions) {
260  input.InsertKnot(pd_query, knot_to_insert);
261  }
262  }
263 
264  // Retrieve id information
265  const std::vector<std::vector<int>>& ids =
266  ExtractBezierPatchIDs(input.SplinepyKnotMultiplicities(), degrees.data());
267 
268  // Number of total patches
269  const int n_total_patches = ids.size();
270  const int n_ctps_per_patch = ids[0].size();
271 
272  // Init return value
273  std::vector<std::shared_ptr<splinepy::splines::SplinepyBase>> bezier_list{};
274  bezier_list.reserve(n_total_patches);
275 
276  // get base control points to copy - in case of NURBS, this will be weighted
277  // coordinates with the weights at the last column
278  const auto& bspline_cps = input.GetCoordinates();
279 
280  // prepare temporary space to copy control point values
281  splinepy::utils::Array<double, 2> extracted_cps(n_ctps_per_patch, dim);
282  splinepy::utils::Array<double> extracted_weights;
283  double* extracted_weights_data{nullptr};
284  if constexpr (is_rational) {
285  extracted_weights.Reallocate(n_ctps_per_patch);
286  extracted_weights_data = extracted_weights.data();
287  }
288 
289  // Loop over the individual patches
290  for (const std::vector<int>& cp_ids : ids) {
291 
292  // loop over cp ids to extract
293  int local_counter{};
294  for (const int& cp_id : cp_ids) {
295  if constexpr (is_rational) {
296  // we have to un-weight
297  // get beginning of from and to row
298  const double* from_ptr = &bspline_cps(cp_id, 0);
299  double* to_ptr = &extracted_cps(local_counter, 0);
300  // invert weight
301  const double w_inv = 1. / from_ptr[dim];
302 
303  // apply inverted weight and save to temporary
304  for (int i{}; i < dim; ++i) {
305  to_ptr[i] = from_ptr[i] * w_inv;
306  }
307  // copy weight
308  extracted_weights[local_counter] = from_ptr[dim];
309 
310  ++local_counter;
311  } else {
312  // for non-rational bspline, we can just copy values
313  std::copy_n(&bspline_cps(cp_id, 0),
314  dim,
315  &extracted_cps(local_counter++, 0));
316  }
317  }
318 
319  // use dynamic spline creator to create beziers.
320  // weights's data will be nullptr if it is non rational.
321  // all the values are copied by bezier and rational bezier
322  bezier_list.push_back(splinepy::splines::SplinepyBase::SplinepyCreate(
323  para_dim,
324  dim,
325  degrees.data(),
326  nullptr,
327  extracted_cps.data(),
328  extracted_weights_data));
329  }
330 
331  return bezier_list;
332 }
