37 std::shared_ptr<SplinepyBase> SplinepyBase::SplinepyCreate(
38  const int para_dim,
39  const int dim,
40  const int* degrees,
41  const std::vector<std::vector<double>>* knot_vectors,
42  double* control_points,
43  double* weights) {
44  if (!degrees || !control_points) {
45  splinepy::utils::PrintAndThrowError(
46  "Not Enough information to create any spline.");
47  }
48 
49  if (!knot_vectors) {
50  if (!weights) {
51  return SplinepyCreateBezier(para_dim, dim, degrees, control_points);
52  } else {
53  return SplinepyCreateRationalBezier(para_dim,
54  dim,
55  degrees,
56  control_points,
57  weights);
58  }
59  } else {
60  if (!weights) {
61  return SplinepyCreateBSpline(para_dim,
62  dim,
63  degrees,
64  knot_vectors,
65  control_points);
66  } else {
67  return SplinepyCreateNurbs(para_dim,
68  dim,
69  degrees,
70  knot_vectors,
71  control_points,
72  weights);
73  }
74  }
75 }
