29 std::vector<std::vector<int>>
30 ExtractBezierPatchIDs(const std::vector<std::vector<int>>& knot_multiplicities,
31  const int* degrees) {
32 
33  const int para_dim = knot_multiplicities.size();
34 
35  // Offsets for start values of individual patches
36  std::vector<std::size_t> bezier_index_offsets{};
37  bezier_index_offsets.reserve(para_dim);
38  bezier_index_offsets.push_back(1);
39  std::vector<std::size_t> control_mesh_resolution(para_dim);
40  std::vector<std::size_t> n_patches_per_para_dim(para_dim);
41 
42  control_mesh_resolution[0] = std::accumulate(knot_multiplicities[0].begin(),
43  knot_multiplicities[0].end(),
44  static_cast<std::size_t>(0))
45  - degrees[0] - 1;
46  n_patches_per_para_dim[0] = knot_multiplicities[0].size() - 1;
47 
48  // Number of total patches and ctps per patch
49  int n_total_patches = n_patches_per_para_dim[0];
50  int n_ctps_per_patch = degrees[0] + 1;
51 
52  for (int i_para_dim{1}; i_para_dim < para_dim; i_para_dim++) {
53  n_patches_per_para_dim[i_para_dim] =
54  knot_multiplicities[i_para_dim].size() - 1;
55  n_total_patches *= n_patches_per_para_dim[i_para_dim];
56  n_ctps_per_patch *= degrees[i_para_dim] + 1;
57  bezier_index_offsets.push_back(bezier_index_offsets[i_para_dim - 1]
58  * (degrees[i_para_dim - 1] + 1));
59  control_mesh_resolution[i_para_dim] =
60  std::accumulate(knot_multiplicities[i_para_dim].begin(),
61  knot_multiplicities[i_para_dim].end(),
62  static_cast<std::size_t>(0))
63  - degrees[i_para_dim] - 1;
64  }
65 
66  // Init return values
67  std::vector<std::size_t> patch_ctp_id_offsets(para_dim, 0);
68  std::vector<std::vector<int>> list_of_id_lists(n_total_patches);
69  std::vector<int> patch_coords(para_dim, 0);
70  for (int i_patch{}; i_patch < n_total_patches; i_patch++) {
71  // Determine internal positions in local coord system
72  int ii{i_patch};
73  // Determine the parameter wise ids of the patch (i.e. the
74  // patch-coordinates) and calculate the required index offsets
75  for (int i{}; i < para_dim; i++) {
76  // ID in spline coordinate system of current patch
77  const int patch_coord = static_cast<int>(ii % n_patches_per_para_dim[i]);
78 
79  // Coordinate offset of the control points indices
80  if (patch_coords[i] != patch_coord) {
81  patch_coords[i] = patch_coord;
82  patch_ctp_id_offsets[i] =
83  (patch_coord == 0)
84  ? 0
85  : patch_ctp_id_offsets[i] + knot_multiplicities[i][patch_coord];
86  }
87 
88  ii -= patch_coord;
89  ii /= n_patches_per_para_dim[i];
90  }
91 
92  // Init vectors required for initialization
93  std::vector<int>& ids = list_of_id_lists[i_patch];
94  ids.reserve(n_ctps_per_patch);
95 
96  // Extract relevant coordinates
97  for (int i_local_id{}; i_local_id < static_cast<int>(n_ctps_per_patch);
98  i_local_id++) {
99  int global_id{};
100  int n_ctps_in_previous_layers{1};
101  // Determine index of local point in global spline
102  for (int i_para_dim{}; i_para_dim < para_dim; i_para_dim++) {
103  // First id in local system
104  const int local_id = (i_local_id / bezier_index_offsets[i_para_dim])
105  % (degrees[i_para_dim] + 1);
106  // Add patch offsets
107  global_id += (local_id + patch_ctp_id_offsets[i_para_dim])
108  * n_ctps_in_previous_layers;
109  // Multiply to index offset
110  n_ctps_in_previous_layers *= control_mesh_resolution[i_para_dim];
111  }
112  ids.push_back(global_id);
113  }
114  }
115  return list_of_id_lists;
116 }
