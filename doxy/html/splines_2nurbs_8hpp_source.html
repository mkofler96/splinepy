116  static Base_ CreateBase(const int* degrees,
117  const std::vector<std::vector<double>>& knot_vectors,
118  double* control_points,
119  double* weights,
120  const int dim) {
121  // process all the info and turn them into SplineLib types to initialize
122  // Base_.
123 
124  // Prepare temporary containers
125  Degrees_ sl_degrees; // std::array
126  Knots_ sl_knots; // std::vector
127  KnotVectors_ sl_knot_vectors; // std::array
128  Coordinates_ sl_control_points; // std::vector
129  Weights_ sl_weights; // std::vector
130  std::size_t ncps{1};
131 
132  // Formulate degrees and knotvectors
133  for (std::size_t i{}; i < kParaDim; ++i) {
134  // degrees
135  sl_degrees[i] = Degree_(degrees[i]);
136  // knot vectors
137  const auto& knot_vector = knot_vectors[i];
138  std::size_t nkv = knot_vector.size();
139  sl_knots.clear();
140  sl_knots.reserve(nkv);
141  // try if this works after namedtype ext
142  // sl_knots = knot_vector;
143  for (std::size_t j{}; j < nkv; ++j) {
144  sl_knots.emplace_back(Knot_{knot_vector[j]});
145  }
146  std::shared_ptr sl_knot_vector{std::make_shared<KnotVector_>(sl_knots)};
147  sl_knot_vectors[i] = sl_knot_vector;
148 
149  ncps *= nkv - degrees[i] - 1;
150  }
151 
152  // Formulate ParameterSpace
153  auto sl_parameter_space =
154  std::make_shared<ParameterSpace_>(sl_knot_vectors, sl_degrees);
155 
156  sl_control_points.SetData(control_points);
157  sl_control_points.SetShape(ncps, dim);
158 
159  sl_weights.SetData(weights);
160  sl_weights.SetShape(ncps);
161 
162  auto sl_weighted_space =
163  std::make_shared<WeightedVectorSpace_>(std::move(sl_control_points),
164  std::move(sl_weights));
165 
166  // return init
167  return Base_(sl_parameter_space, sl_weighted_space);
168  }

247  virtual void
248  SplinepyCurrentProperties(int* degrees,
249  std::vector<std::vector<double>>* knot_vectors,
250  double* control_points,
251  double* weights) const {
252  const auto& parameter_space = GetParameterSpace();
253  const auto& vector_space = GetWeightedVectorSpace();
254  const int dim = Base_::Dim();
255 
256  // degrees
257  if (degrees) {
258  for (int i{}; i < kParaDim; ++i) {
259  degrees[i] = static_cast<int>(parameter_space.GetDegrees()[i]);
260  }
261  }
262 
263  // knot_vectors
264  if (knot_vectors) {
265  const auto& core_kvs = parameter_space.GetKnotVectors();
266  knot_vectors->clear();
267  knot_vectors->reserve(kParaDim);
268  for (int i{}; i < kParaDim; ++i) {
269  const auto& core_kv = *core_kvs[i];
270  const int kvsize = core_kv.GetSize();
271  std::vector<double> kv;
272  kv.reserve(kvsize);
273  // Index wants int
274  for (int j{}; j < kvsize; ++j) {
275  kv.emplace_back(static_cast<double>(core_kv[bsplinelib::Index{j}]));
276  }
277  knot_vectors->push_back(std::move(kv));
278  }
279  }
280 
281  // control_points and weights
282  if (control_points || weights) {
283  int ncps = vector_space.GetNumberOfCoordinates();
284  // fill it up, phil!
285  for (int i{}; i < ncps; ++i) {
286  const auto* coord_begin = vector_space.CoordinateBegin(i);
287 
288  // unweight - phil needs to first project before filling.
289  if (control_points) {
290  const double w_inv = 1. / coord_begin[dim];
291  for (int j{}; j < dim; ++j) {
292  control_points[i * dim + j] = coord_begin[j] * w_inv;
293  }
294  }
295  if (weights) {
296  weights[i] = coord_begin[dim];
297  }
298  }
299  }
300  }

317  virtual std::shared_ptr<WeightedControlPointPointers_>
318  SplinepyWeightedControlPointPointers() {
319  if (SplinepyBase_::control_point_pointers_
320  && SplinepyBase_::control_point_pointers_->Len()
321  == SplinepyNumberOfControlPoints()) {
322  return SplinepyBase_::control_point_pointers_;
323  }
324 
325  // get sizes
326  const int dim = Base_::Dim();
327  const int n_cps = SplinepyNumberOfControlPoints();
328 
329  // create weighted cps
330  auto wcpp = std::make_shared<WeightedControlPointPointers_>();
331  wcpp->dim_ = dim;
332  wcpp->for_rational_ = kIsRational;
333  wcpp->coordinate_begins_.resize(n_cps);
334  // create weights
335  auto w = std::make_shared<WeightPointers_>();
336  w->weights_.resize(n_cps);
337 
338  auto& w_control_points = Base_::weighted_vector_space_->GetCoordinates();
339  for (int i{}; i < n_cps; ++i) {
340  // get beginning of each coordinate
341  auto* coord_begin = &w_control_points(i, 0);
342  // save each coords
343  wcpp->coordinate_begins_[i] = coord_begin;
344  w->weights_[i] = coord_begin + dim;
345  }
346 
347  // reference each other
348  w->control_point_pointers_ = wcpp; // weak-ref
349  wcpp->weight_pointers_ = w;
350 
351  // save
352  SplinepyBase_::control_point_pointers_ = wcpp;
353 
354  return wcpp;
355  }
