36 std::tuple<py::array_t<double>, std::vector<int>>
37 ComputeKnotInsertionMatrixAndKnotSpan(const py::array_t<double>& old_kv,
38  const py::array_t<double>& new_kv,
39  const int degree,
40  const double& tolerance) {
41  // Auxiliary access values
42  const double* old_kv_ptr = static_cast<double*>(old_kv.request().ptr);
43  const double* new_kv_ptr = static_cast<double*>(new_kv.request().ptr);
44  const int n_cps_old{static_cast<int>(old_kv.size()) - degree - 1};
45  const int n_cps_new{static_cast<int>(new_kv.size()) - degree - 1};
46  // Create and initialize
47  py::array_t<double> return_matrix(n_cps_new * n_cps_old);
48  double* return_matrix_ptr = static_cast<double*>(return_matrix.request().ptr);
49  for (int i{}; i < n_cps_new * n_cps_old; i++) {
50  return_matrix_ptr[i] = 0.;
51  }
52 
53  // Helper function, because I can't be bothered
54  auto matrix = [&](const int& i, const int& j) -> double& {
55  return return_matrix_ptr[i * n_cps_old + j];
56  };
57 
58  // Helper function to create specific entries of the B-Spline Matrix on
59  // the fly (as they are only required a single time)
60  auto R_matrix = [&](const double& t, // Evaluation point
61  const int& mu, // Offset with regards to local support
62  const int& deg, // Matrix degree
63  const int& i, // row
64  const int& j // col
65  ) -> double {
66  // Safety checks (Might be put into debugging block)
67  if ((i >= deg) || j > deg) {
68  splinepy::utils::PrintAndThrowError(
69  "Requested index outside matrix definition.",
70  "Requested query: \nt :",
71  t,
72  "\nmu : ",
73  mu,
74  "\ndeg : ",
75  deg,
76  "\ni : ",
77  i,
78  "\nj : ",
79  j);
80  }
81  // Matrix is bi-diagonal i == j and i == j+1
82  if ((i > j) || (j > i + 1)) {
83  return 0.;
84  }
85 
86  // Actual values
87  if (j == i) {
88  return (old_kv_ptr[mu + 1 + i] - t)
89  / (old_kv_ptr[mu + 1 + i] - old_kv_ptr[mu + 1 + i - deg]);
90  } else {
91  return (t - old_kv_ptr[mu + 1 + i - deg])
92  / (old_kv_ptr[mu + 1 + i] - old_kv_ptr[mu + 1 + i - deg]);
93  }
94  };
95 
96  // Another auxiliary values
97  std::vector<int> following_n_equal_knots(new_kv.size());
98  std::vector<int> index_of_knot_span_base(new_kv.size());
99 
100  // As we assume closed knot-vectors (we can ignore the last entry)
101  int current_old_id{static_cast<int>(old_kv.size() - 2)};
102  int current_knot_span{static_cast<int>(old_kv.size() - 1)};
103 
104  // Prepare some values that will be required later (Ignore last value loop
105  // begins at new_kv.size() - 2)
106  for (int i{static_cast<int>(new_kv.size() - 1)}; i-- > 0;) {
107  // Check knots against each other with a tolerance
108  if (std::abs(old_kv_ptr[current_old_id] - new_kv_ptr[i]) < tolerance) {
109  // Check if we entered another non-zero knot-span
110  if (old_kv_ptr[current_old_id] < old_kv_ptr[current_old_id + 1]) {
111  current_knot_span = current_old_id;
112  }
113  current_old_id--;
114  } else {
115  if (old_kv_ptr[current_old_id] > new_kv_ptr[i]) {
116  // Beautify the output
117  std::ostringstream old_s, new_s;
118  new_s << "[" << new_kv_ptr[0];
119  for (int i_kv{1}; i_kv < new_kv.size(); i_kv++) {
120  new_s << ", " << new_kv_ptr[i];
121  }
122  new_s << "]";
123  old_s << "[" << old_kv_ptr[0];
124  for (int i_kv{1}; i_kv < old_kv.size(); i_kv++) {
125  old_s << ", " << old_kv_ptr[i];
126  }
127  old_s << "]";
128  splinepy::utils::PrintAndThrowError(
129  "New knot spans is not subset of old knot-span. \nOld knot vector "
130  ": ",
131  old_s.str(),
132  "\nNew knot vector : ",
133  new_s.str());
134  }
135  }
136  if (new_kv_ptr[i] < old_kv_ptr[current_knot_span]) {
137  index_of_knot_span_base[i] = current_old_id;
138  } else {
139  index_of_knot_span_base[i] = current_knot_span;
140  }
141  }
142 
143  // Start assigning values to the matrix
144  int offset{0};
145 
146  for (int i{0}; i < n_cps_new; i++) {
147  // Get mu from previous knot_span
148  const int mu{index_of_knot_span_base[i]};
149 
150  // Get to the first non-zero entry within the row
151  int j{mu - degree};
152 
153  // Check if identity can be applied
154  if (following_n_equal_knots[i + 1] > degree) {
155  matrix(i, i - offset) = 1;
156  // The degree == 0 case is also caught with this
157  continue;
158  }
159 
160  // Increase offset for every new knot
161  if (following_n_equal_knots[i + 1] == 0) {
162  offset++;
163  }
164 
165  // As degree > 0 it must be at least 1 so we can pre-assign the first degree
166  // values and write them into the matrix
167  matrix(i, j) = R_matrix(new_kv_ptr[i + 1], mu, 1, 0, 0);
168  matrix(i, j + 1) = R_matrix(new_kv_ptr[i + 1], mu, 1, 0, 1);
169 
170  // Loop over all remaining matrices, use their sparsity to avoid unnecessary
171  // computations and increase degree to be more efficient (no double matrix
172  // computations)
173  for (int d{2}; d <= degree; d++) {
174  // Loop over different matrices, last and first column of the d matrix
175  // must be treated differently, because they only have one entry, we loop
176  // backwards so we can avoid creating an additional auxiliary vectors
177  matrix(i, j + d) =
178  (R_matrix(new_kv_ptr[i + d], mu, d, d - 1, d) * matrix(i, j + d - 1));
179  // For matrix multiplication we can go backwards to avoid creating an
180  // additional aux vector and store solution of multiplication directly
181  // into the solution matrix. A little harder to read, but makes the code
182  // more efficient.
183  for (int q{d - 1}; q > 0; q--) {
184  matrix(i, j + q) =
185  matrix(i, j + q - 1) * R_matrix(new_kv_ptr[i + d], mu, d, q - 1, q)
186  + matrix(i, j + q) * R_matrix(new_kv_ptr[i + d], mu, d, q, q);
187  }
188  // Assign first column last to complete matrix vector product for given
189  // line
190  matrix(i, j) = R_matrix(new_kv_ptr[i + d], mu, d, 0, 0) * matrix(i, j);
191  }
192  }
193  // assign shape and return result
194  return_matrix.resize({(int) n_cps_new, (int) n_cps_old});
195  return std::make_tuple(return_matrix, index_of_knot_span_base);
196 }

207 py::tuple ComputeGlobalKnotInsertionMatrix(
208  const std::vector<py::array_t<double>>& old_kvs,
209  const py::array_t<int>& degrees,
210  const int parametric_dimension,
211  const py::array_t<double>& new_knots,
212  const double& tolerance) {
213 
214  // Create aliases
215  const int* degrees_ptr = static_cast<int*>(degrees.request().ptr);
216  const int n_para_dims = static_cast<int>(degrees.size());
217  double* new_knots_ptr = static_cast<double*>(new_knots.request().ptr);
218 
219  // Make sure new knots are sorted
220  std::sort(new_knots_ptr, new_knots_ptr + new_knots.size());
221 
222  // Precompute and initialize auxiliary values
223  std::vector<int> n_cp_old(n_para_dims), n_cp_new(n_para_dims);
224  int n_cps_old{1};
225  for (int i{}; i < n_para_dims; i++) {
226  // Knot vectors are stored as numpy arrays
227  const py::array_t<double>& knot_vector_old = old_kvs[i];
228 
229  // Comput number of cps before and after insertion
230  n_cp_old[i] = knot_vector_old.size() - degrees_ptr[i] - 1;
231  n_cp_new[i] = n_cp_old[i];
232  n_cps_old *= n_cp_old[i];
233  }
234  n_cp_new[parametric_dimension] += new_knots.size();
235  const int n_cps_new = n_cps_old / n_cp_old[parametric_dimension]
236  * n_cp_new[parametric_dimension];
237 
238  // Compute dimensionwise (or line-) matrices. These matrices
239  // are very small compared to higher dimensional spline matrices.
240  const py::array_t<double>& knot_vector_old = old_kvs[parametric_dimension];
241  double* knot_vector_old_ptr =
242  static_cast<double*>(knot_vector_old.request().ptr);
243 
244  // Build new knot_vector
245  py::array_t<double> knot_vector_new(knot_vector_old.size()
246  + new_knots.size());
247  double* knot_vector_new_ptr =
248  static_cast<double*>(knot_vector_new.request().ptr);
249  for (int i_old{}, j_new{}; i_old + j_new < knot_vector_new.size();) {
250  if ((knot_vector_old_ptr[i_old] < new_knots_ptr[j_new])
251  || (j_new >= new_knots.size())) {
252  knot_vector_new_ptr[i_old + j_new] = knot_vector_old_ptr[i_old];
253  i_old++;
254  } else {
255  knot_vector_new_ptr[i_old + j_new] = new_knots_ptr[j_new];
256  j_new++;
257  }
258  }
259 
260  // Compute dimensionwise (or line-) matrices. These matrices
261  // are very small compared to higher dimensional spline matrices.
262  const auto results =
263  ComputeKnotInsertionMatrixAndKnotSpan(knot_vector_old,
264  knot_vector_new,
265  degrees_ptr[parametric_dimension],
266  tolerance);
267  const py::array_t<double> dimensionwise_matrix{std::get<0>(results)};
268  const double* dimensionwise_matrix_ptr =
269  static_cast<double*>(dimensionwise_matrix.request().ptr);
270  const std::vector<int> knot_span_ids{std::get<1>(results)};
271 
272  // Create auxiliary functions to access individual cps positions
273  auto local_to_global_id = [&](const std::vector<int>& local_id,
274  const std::vector<int>& n_cps) -> int {
275  int global_id{local_id[n_para_dims - 1]};
276  for (int i{n_para_dims - 1}; i-- > 0;) {
277  global_id *= n_cps[i + 0];
278  global_id += local_id[i];
279  }
280  return global_id;
281  };
282  auto global_to_local_id =
283  [&](const int& global_id,
284  const std::vector<int>& n_cps) -> std::vector<int> {
285  // Init
286  std::vector<int> local_ids{};
287  int copy_{global_id};
288  local_ids.reserve(n_para_dims);
289  for (int i{}; i < n_para_dims; i++) {
290  const int r = copy_ % n_cps[i];
291  local_ids.push_back(r);
292  copy_ /= n_cps[i];
293  }
294  return local_ids;
295  };
296 
297  // All matrices have specific size n_cps_new (post knot insertion) x
298  // n_cps_old (prior to knot insertion), with a k-diagonal block structure,
299  // where k is the degree along the parametric axis i. Some of the values
300  // will be zero, if they do not contribute to one of the new inserted knots.
301 
302  // Initiate return values
303  const int n_entries{(degrees_ptr[parametric_dimension] + 1) * n_cps_new};
304  py::array_t<double> values(n_entries);
305  py::array_t<int> rows(n_entries);
306  py::array_t<int> cols(n_entries);
307 
308  // Retrieve pointers to data
309  double* values_ptr = static_cast<double*>(values.request().ptr);
310  int* rows_ptr = static_cast<int*>(rows.request().ptr);
311  int* cols_ptr = static_cast<int*>(cols.request().ptr);
312 
313  // Assign values accordingly
314  int counter{};
315  for (int j_ctps{}; j_ctps < n_cps_new; ++j_ctps) {
316  // j_ctps refers to the row associated to the current knot-span basis
317  // n_cp_current
318  std::vector<int> local_id_new = global_to_local_id(j_ctps, n_cp_new);
319  const int i_row = local_id_new[parametric_dimension];
320  // get knot-span
321  const int knot_span_base_id =
322  knot_span_ids[local_id_new[parametric_dimension]];
323 
324  for (int k_deg{knot_span_base_id - degrees_ptr[parametric_dimension]};
325  k_deg <= knot_span_base_id;
326  ++k_deg) {
327  local_id_new[parametric_dimension] = k_deg;
328  const int col_id = local_to_global_id(local_id_new, n_cp_old);
329  // Assign values
330  rows_ptr[counter] = static_cast<int>(j_ctps);
331  cols_ptr[counter] = static_cast<int>(col_id);
332  values_ptr[counter] =
333  dimensionwise_matrix_ptr[i_row * n_cp_old[parametric_dimension]
334  + k_deg];
335  counter++;
336  }
337  }
338 
339  // Add contributions to list
340  return py::make_tuple(
341  py::make_tuple(values, py::make_tuple(rows, cols)),
342  py::make_tuple(static_cast<int>(n_cps_new), static_cast<int>(n_cps_old)));
343 }

345 py::tuple BezierExtractionMatrices(const std::shared_ptr<PySpline>& spline,
346  const double& tolerance) {
347  // Create property copies
348  const int n_para_dims = spline->Core()->SplinepyParaDim();
349  std::vector<int> degrees(n_para_dims);
350  std::vector<std::vector<double>> old_kvs;
351  spline->Core()->SplinepyCurrentProperties(degrees.data(),
352  &old_kvs,
353  nullptr,
354  nullptr);
355 
356  // Given a knot vector function provides a new knot vector in python format
357  // that can represents the c^(-1) links
358  auto create_new_knot_vector =
359  [&](const py::array_t<double>& old_kv,
360  const int& deg,
361  py::array_t<double>& new_knots) -> py::array_t<double> {
362  // allocate maximum required space for initialization
363  const int old_kv_size = old_kv.size();
364  const int max_kv_size = (old_kv_size - (deg + 1) * 2) * deg + (deg + 1) * 2;
365  py::array_t<double> return_kv(max_kv_size);
366  double* return_kv_ptr = static_cast<double*>(return_kv.request().ptr);
367  double* old_kv_ptr = static_cast<double*>(old_kv.request().ptr);
368  double* new_knots_ptr = static_cast<double*>(new_knots.request().ptr);
369  int i_k_new{}, i_k_old{}, ref_count{};
370  double c_knot{old_kv_ptr[0]};
371  while (i_k_old < old_kv_size) {
372  // If a jump to a new knot vector is a valid choice, we can update our
373  // current knot reference
374  if (ref_count > deg) {
375  c_knot = old_kv_ptr[i_k_old];
376  ref_count = 1;
377  }
378 
379  if (std::abs(c_knot - old_kv_ptr[i_k_old]) < tolerance) {
380  // Knot vector has not changed
381  return_kv_ptr[i_k_new] = c_knot;
382  ref_count++;
383  i_k_old++;
384  i_k_new++;
385  } else {
386  // Knot vector has changed
387  return_kv_ptr[i_k_new] = c_knot;
388  new_knots_ptr[i_k_new - i_k_old] = c_knot;
389  i_k_new++;
390  ref_count++;
391  }
392  }
393  return_kv.resize({static_cast<int>(i_k_new)});
394  new_knots.resize({static_cast<int>(i_k_new - i_k_old)});
395  return return_kv;
396  };
397 
398  // Initialize the return value
399  py::list list_of_tuples{};
400  // list of kv references - used to call ComputeGlobalKnotInsertionMatrix
401  // we start with original kvs and this changes as we virtually keep inserting
402  // knots each dimension
403  std::vector<py::array_t<double>> tmp_kvs;
404  tmp_kvs.reserve(n_para_dims);
405  for (auto& kv : old_kvs) {
406  // create views of old kvs as initial elements
407  tmp_kvs.push_back(py::array_t<double>(kv.size(), kv.data(), py::none()));
408  }
409 
410  // Here we compute the individual components that are required to initialize
411  // the sparse matrices in the python frontend
412  // numpy view to call ComputeGlobalKnotInsertionMatrix
413  py::array_t<int> np_degrees(degrees.size(), degrees.data(), py::none());
414  for (int parametric_dimension{}; parametric_dimension < n_para_dims;
415  parametric_dimension++) {
416  const py::array_t<double>& tmp_kv = tmp_kvs[parametric_dimension];
417  const int max_new_knots =
418  (tmp_kv.size() - (degrees[parametric_dimension] + 1) * 2)
419  * degrees[parametric_dimension];
420 
421  py::array_t<double> new_knots(max_new_knots);
422  const auto& new_kv = create_new_knot_vector(tmp_kv,
423  degrees[parametric_dimension],
424  new_knots);
425 
426  // Add data to list
427  list_of_tuples.append(ComputeGlobalKnotInsertionMatrix(tmp_kvs,
428  np_degrees,
429  parametric_dimension,
430  new_knots,
431  tolerance));
432  tmp_kvs[parametric_dimension] = new_kv;
433  }
434 
435  // to extract bezier ids, create knot multiplicity info.
436  std::vector<std::vector<int>> knot_multiplicities;
437  knot_multiplicities.reserve(n_para_dims);
438  for (const py::array_t<double>& kv : tmp_kvs) {
439  knot_multiplicities.emplace_back(
440  bsplinelib::parameter_spaces::KnotVector::DetermineMultiplicities(
441  static_cast<double*>(kv.request().ptr),
442  kv.size(),
443  tolerance));
444  }
445 
446  // Lastly, compute the bezier extraction points, that correspond to the global
447  // matrix.
448  const auto& list_of_ids =
449  splines::helpers::ExtractBezierPatchIDs(knot_multiplicities,
450  degrees.data());
451  const int n_patches = list_of_ids.size();
452  const int n_ctps_per_patch = list_of_ids[0].size();
453  py::array_t<int> bezier_ctps_ids(n_ctps_per_patch * n_patches);
454  int* bezier_ctps_ids_ptr = static_cast<int*>(bezier_ctps_ids.request().ptr);
455  for (int i_patch{}; i_patch < n_patches; i_patch++) {
456  for (int j_id{}; j_id < n_ctps_per_patch; j_id++) {
457  bezier_ctps_ids_ptr[i_patch * n_ctps_per_patch + j_id] =
458  list_of_ids[i_patch][j_id];
459  }
460  }
461 
462  bezier_ctps_ids.resize(
463  {static_cast<int>(n_patches), static_cast<int>(n_ctps_per_patch)});
464  return py::make_tuple(bezier_ctps_ids, list_of_tuples);
465 }
