33 py::array_t<bool> InsertKnots(std::shared_ptr<PySpline>& spline,
34  int para_dim,
35  py::array_t<double> knots) {
36  double* knots_ptr = static_cast<double*>(knots.request().ptr);
37  const int n_request = knots.size();
38 
39  // let's argsort, get unique, get multiplicity
40  splinepy::utils::Array<double, 1, int> knot_request_view(knots_ptr,
41  n_request);
42  splinepy::utils::Array<int, 1, int> arg_sorted, unique, multiplicity;
43  knot_request_view.ArgSort(arg_sorted);
44  splinepy::utils::UniqueIndicesAndMultiplicities(knot_request_view,
45  arg_sorted,
46  unique,
47  multiplicity);
48 
49  py::array_t<bool> successful(n_request);
50  bool* successful_ptr = static_cast<bool*>(successful.request().ptr);
51  const int* arg_ptr = arg_sorted.begin();
52  for (int i{}; i < unique.size(); ++i) {
53  int n_added = spline->Core()->SplinepyInsertKnot(para_dim,
54  knots_ptr[unique[i]],
55  multiplicity[i]);
56  // sets results
57  for (int j{}; j < multiplicity[i]; ++j) {
58  successful_ptr[*arg_ptr++] = (n_added-- > 0);
59  }
60  }
61 
62  return successful;
63 }

125 py::list Split(const std::shared_ptr<PySpline>& spline,
126  int p_dim,
127  py::array_t<double> locations) {
128  // make sure they are sorted
129  std::vector<double> locs(locations.size());
130  std::copy_n(static_cast<double*>(locations.request().ptr),
131  locations.size(),
132  locs.begin());
133  // sort
134  std::sort(locs.begin(), locs.end());
135 
136  // split and append
137  py::list split;
138  // very first
139  auto tmp_splitted = spline->Core()->SplinepySplit(p_dim, locs[0]);
140  split.append(std::make_shared<PySpline>(tmp_splitted[0]));
141  for (std::size_t i{1}; i < locs.size(); ++i) {
142  const double locs_with_offset =
143  (locs[i] - locs[i - 1]) / (1. - locs[i - 1]);
144  tmp_splitted = tmp_splitted[1]->SplinepySplit(p_dim, locs_with_offset);
145  split.append(std::make_shared<PySpline>(tmp_splitted[0]));
146  }
147  // very last
148  split.append(std::make_shared<PySpline>(tmp_splitted[1]));
149 
150  return split;
151 }

164 py::list ExtractBoundaries(const std::shared_ptr<PySpline>& spline,
165  const py::array_t<int>& boundary_ids) {
166  // Init return value
167  py::list boundary_splines{};
168  const int n_boundaries = boundary_ids.size();
169  int* bid_ptr = static_cast<int*>(boundary_ids.request().ptr);
170  if (boundary_ids.size() == 0) {
171  for (int i{}; i < spline->para_dim_ * 2; ++i) {
172  boundary_splines.append(std::make_shared<PySpline>(
173  spline->Core()->SplinepyExtractBoundary(i)));
174  }
175  } else {
176  const int max_bid = spline->para_dim_ * 2 - 1;
177  for (int i{}; i < n_boundaries; ++i) {
178  const int& bid = bid_ptr[i];
179  if (bid < 0 || bid > max_bid) {
180  splinepy::utils::PrintAndThrowError("Requested Boundary ID :",
181  bid,
182  "exceeds admissible range.");
183  }
184  boundary_splines.append(std::make_shared<PySpline>(
185  spline->Core()->SplinepyExtractBoundary(bid_ptr[i])));
186  }
187  }
188 
189  return boundary_splines;
190 }

212 std::shared_ptr<PySpline>
213 SameSplineWithKnotVectors(std::shared_ptr<PySpline>& spline) {
214  // early exit if the spline has knot vectors already
215  if (spline->HasKnotVectors()) {
216  return spline;
217  }
218 
219  py::dict props = spline->CurrentCoreProperties();
220 
221  // based on degrees, generate knot vectors
222  py::array_t<int> degrees = py::cast<py::array_t<int>>(props["degrees"]);
223  int* degrees_ptr = static_cast<int*>(degrees.request().ptr);
224  py::list kvs;
225  for (int i{}; i < degrees.size(); ++i) {
226  // prepare kv and get required number of repeating knots
227  const int n_knot_repeat = degrees_ptr[i] + 1;
228  py::array_t<double> kv(n_knot_repeat * 2);
229  double* kv_ptr = static_cast<double*>(kv.request().ptr);
230 
231  // defined knots with 0 and 1
232  for (int j{}; j < n_knot_repeat; ++j) {
233  kv_ptr[j] = 0.;
234  kv_ptr[n_knot_repeat + j] = 1.;
235  }
236 
237  kvs.append(kv);
238  }
239 
240  // update knot vectors to dict spline
241  props["knot_vectors"] = kvs;
242 
243  return std::make_shared<PySpline>(props);
244 }
