124  static Base_ CreateBase(const int* degrees,
125  const std::vector<std::vector<double>>& knot_vectors,
126  double* control_points,
127  const int dim) {
128  // process all the info and turn them into SplineLib types to initialize
129  // Base_.
130 
131  // Prepare temporary containers
132  Degrees_ sl_degrees; // std::array
133  Knots_ sl_knots; // std::vector
134  KnotVectors_ sl_knot_vectors; // std::array
135  Coordinates_ sl_control_points;
136  int ncps{1};
137 
138  // Formulate degrees and knotvectors
139  for (int i{}; i < kParaDim; ++i) {
140  // degrees
141  sl_degrees[i] = Degree_(degrees[i]);
142  // knot vectors
143  const auto& knot_vector = knot_vectors[i];
144  const int nkv = knot_vector.size();
145  sl_knots.clear();
146  sl_knots.reserve(nkv);
147  // try if this works after namedtype ext
148  // sl_knots = knot_vector;
149  for (int j{}; j < nkv; ++j) {
150  sl_knots.emplace_back(Knot_{knot_vector[j]});
151  }
152  std::shared_ptr sl_knot_vector{std::make_shared<KnotVector_>(sl_knots)};
153  sl_knot_vectors[i] = sl_knot_vector;
154 
155  ncps *= nkv - degrees[i] - 1;
156  }
157 
158  // Formulate ParameterSpace
159  auto sl_parameter_space =
160  std::make_shared<ParameterSpace_>(sl_knot_vectors, sl_degrees);
161 
162  // Formulate control_points - this will be a view. make sure to keep the
163  // pointer alive
164  sl_control_points.SetData(control_points);
165  sl_control_points.SetShape(ncps, dim);
166 
167  auto sl_vector_space =
168  std::make_shared<VectorSpace_>(std::move(sl_control_points));
169 
170  // return init
171  return Base_(sl_parameter_space, sl_vector_space);
172  }

240  virtual void
241  SplinepyCurrentProperties(int* degrees,
242  std::vector<std::vector<double>>* knot_vectors,
243  double* control_points,
244  double* weights /* untouched */) const {
245 
246  const auto& parameter_space = GetParameterSpace();
247  const auto& vector_space = GetVectorSpace();
248 
249  // degrees
250  if (degrees) {
251  for (std::size_t i{}; i < kParaDim; ++i) {
252  degrees[i] = static_cast<int>(parameter_space.GetDegrees()[i]);
253  }
254  }
255 
256  // knot_vectors
257  if (knot_vectors) {
258  const auto& core_kvs = parameter_space.GetKnotVectors();
259  knot_vectors->clear();
260  knot_vectors->reserve(kParaDim);
261  for (std::size_t i{}; i < kParaDim; ++i) {
262  const auto& core_kv = *core_kvs[i];
263  const std::size_t kvsize = static_cast<std::size_t>(core_kv.GetSize());
264  std::vector<double> kv;
265  kv.reserve(kvsize);
266  // Index wants int
267  for (int j{}; j < static_cast<int>(kvsize); ++j) {
268  kv.emplace_back(static_cast<double>(core_kv[bsplinelib::Index{j}]));
269  }
270  knot_vectors->push_back(std::move(kv));
271  }
272  }
273 
274  // control_points
275  if (control_points) {
276  // contiguous!
277  const auto& coords = GetCoordinates();
278  std::copy(coords.begin(), coords.end(), control_points);
279  }
280  }
