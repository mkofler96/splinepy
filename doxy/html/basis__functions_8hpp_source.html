113 template<typename SplineType, typename QueryType>
114 inline std::vector<int> BSplineSupport(const SplineType& spline,
115  const QueryType* para_coord) {
116  static_assert(SplineType::kHasKnotVectors,
117  "BSplineBasis is only for bspline families.");
118 
119  // prepare properties for the loop
120  const auto& parameter_space = spline.GetParameterSpace();
121  const auto n_basis_per_dim =
122  parameter_space.GetNumberOfNonZeroBasisFunctions();
123  const auto support_start =
124  parameter_space.FindFirstNonZeroBasisFunction(para_coord);
125  auto support_offset = parameter_space.First();
126  const int n_total_basis =
127  std::accumulate(n_basis_per_dim.begin(),
128  n_basis_per_dim.end(),
129  typename decltype(n_basis_per_dim)::value_type{1},
130  std::multiplies{});
131 
132  // prepare return
133  std::vector<int> supports;
134  supports.reserve(n_total_basis);
135 
136  // fill supports.
137  for (int i{}; i < n_total_basis; ++i) {
138  supports.push_back(
139  (support_start + support_offset.GetIndex()).GetIndex1d());
140  ++support_offset;
141  }
142  return supports;
143 }

168 template<typename SplineType, typename QueryType, typename SupportType>
169 constexpr inline std::vector<QueryType>
170 RationalBSplineBasis(const SplineType& spline,
171  const QueryType* para_coord,
172  const SupportType& support) {
173  static_assert(SplineType::kIsRational && SplineType::kHasKnotVectors,
174  "RationalBSplineBasis is only applicable to NURBS.");
175  const auto bspline_basis = NonRationalBSplineBasis(spline, para_coord);
176  const auto& homogeneous_coords = spline.GetCoordinates();
177  const auto n_basis = support.size();
178  const int dim = spline.SplinepyDim();
179  // prepare output
180  std::vector<QueryType> rational_basis;
181  rational_basis.reserve(n_basis);
182 
183  double W{0.};
184  for (std::size_t i{}; i < n_basis; ++i) {
185  // get weight
186  const auto& w = homogeneous_coords(support[i], dim);
187  const auto N_times_w = bspline_basis[i] * w;
188 
189  W += N_times_w;
190  rational_basis.push_back(N_times_w);
191  }
192 
193  const auto W_inv = 1 / W;
194  for (auto& rb : rational_basis) {
195  rb *= W_inv;
196  }
197 
198  return rational_basis;
199 }

256 template<typename SplineType, typename QueryType, typename OrderType>
257 inline bsplinelib::parameter_spaces::BasisValues
258 RationalBSplineBasisDerivative(const SplineType& spline,
259  const QueryType* para_coord,
260  const OrderType* order) {
261  using BasisValues = bsplinelib::parameter_spaces::BasisValues;
262 
263  // we will do everything with OrderType
264  constexpr auto para_dim = static_cast<OrderType>(SplineType::kParaDim);
265 
266  // prepare
267  const auto& homogeneous_coords = spline.GetCoordinates();
268  const int dim = homogeneous_coords.Shape()[1] - 1;
269 
270  // Define lambdas to switch between local indexing with
271  // coordinate style indexing to global, scalar indexing
272  // example: for nth (2,1,4) : (1,0,0)->1, (1,1,0)->4
273 
274  // Global (scalar) indexing to local index-system
275  auto local_ids_ =
276  [&](const OrderType req_id) -> std::array<OrderType, para_dim> {
277  OrderType id{req_id};
278  std::array<OrderType, para_dim> local_ids{};
279  for (OrderType i_pd{}; i_pd < para_dim; ++i_pd) {
280  if (order[i_pd] == 0)
281  continue;
282  local_ids[i_pd] = id % (order[i_pd] + 1);
283  id -= local_ids[i_pd];
284  id /= order[i_pd] + 1;
285  }
286  return local_ids;
287  };
288 
289  // Local (coordinate-style) indexing to global
290  auto global_ids_ = [&](const OrderType* req_derivs) -> OrderType {
291  OrderType id{};
292  OrderType offset{1};
293  for (OrderType i_pd{}; i_pd < para_dim; ++i_pd) {
294  // assert(req_derivs[i_pd] <= order[i_pd]);
295  if (order[i_pd] == 0)
296  continue;
297  id += offset * (req_derivs[i_pd]);
298  offset = order[i_pd] > 0 ? offset * (order[i_pd] + 1) : offset;
299  }
300  return id;
301  };
302 
303  // Check if requested derivative is "subset" to current derivative
304  auto is_not_subset_ =
305  [&](const std::array<OrderType, para_dim>& req_derivs_max,
306  const std::array<OrderType, para_dim>& req_derivs) -> bool {
307  for (OrderType i_pd{}; i_pd < para_dim; ++i_pd) {
308  if (req_derivs[i_pd] > req_derivs_max[i_pd])
309  return true;
310  }
311  return false;
312  };
313 
314  // Prepare supports
315  const auto supports = BSplineSupport(spline, para_coord);
316  // Initialize return type
317  const OrderType number_of_derivs{global_ids_(order) + 1};
318  const OrderType n_basis_functions{spline.SplinepyNumberOfSupports()};
319  // Please remember that the first derivative is not used
320  std::vector<BasisValues> derivatives(number_of_derivs);
321  std::vector<BasisValues> A_derivatives(number_of_derivs);
322  BasisValues w_derivatives(number_of_derivs);
323  w_derivatives.Fill(0.);
324 
325  // Fill all polynomial spline derivatives (and values for id=0)
326  for (OrderType i_deriv{}; i_deriv < number_of_derivs; ++i_deriv) {
327  const auto req_derivs = local_ids_(i_deriv);
328  auto& A_derivatives_i = A_derivatives[i_deriv];
329  A_derivatives_i =
330  std::move(NonRationalBSplineBasisDerivative(spline,
331  para_coord,
332  req_derivs.data()));
333  auto& w_derivatives_i = w_derivatives[i_deriv];
334  for (OrderType i_basis{}; i_basis < n_basis_functions; ++i_basis) {
335  const QueryType weight = homogeneous_coords(supports[i_basis], dim);
336  w_derivatives_i += weight * A_derivatives_i[i_basis];
337  A_derivatives_i[i_basis] *= weight;
338  }
339  }
340 
341  // Precompute inverse of weighted function
342  const QueryType inv_w_fact = static_cast<QueryType>(1.) / w_derivatives[0];
343 
344  // Loop over all lower-order derivatives and assign derivatives-vector
345  // Notation follows "The NURBS book" eq. 4.20 (extended for n-d splines)
346  for (OrderType i_deriv{0}; i_deriv < number_of_derivs; ++i_deriv) {
347  // Retrieve index-wise order of the derivative for current ID
348  const auto derivative_order_indexwise_LHS = local_ids_(i_deriv);
349  // Assign derivative of Numerator-function
350  derivatives[i_deriv].OwnCopy(A_derivatives[i_deriv]);
351  // Subtract all weighted lower-order functions
352  for (OrderType j_deriv{1}; j_deriv <= i_deriv; ++j_deriv) {
353  // Retrieve order of current index
354  const auto derivative_order_indexwise_RHS = local_ids_(j_deriv);
355  // Check only subsets // Define lambdas to switch between local indexing
356  // with coordinate style
357  if (is_not_subset_(derivative_order_indexwise_LHS,
358  derivative_order_indexwise_RHS))
359  continue;
360  // Precompute Product of binomial coefficients
361  // for now, use bezman's implementation - TODO: use SplineLib
362  int binom_fact{1};
363  for (OrderType i_pd{}; i_pd < para_dim; ++i_pd) {
364  binom_fact *=
365  bsplinelib::utilities::math_operations::ComputeBinomialCoefficient(
366  derivative_order_indexwise_LHS[i_pd],
367  derivative_order_indexwise_RHS[i_pd]);
368  }
369  const QueryType binom = static_cast<QueryType>(binom_fact);
370  // Subtract low-order function
371  for (OrderType i_basis{}; i_basis < n_basis_functions; ++i_basis) {
372  derivatives[i_deriv][i_basis] -=
373  binom * w_derivatives[j_deriv]
374  * derivatives[i_deriv - j_deriv][i_basis];
375  }
376  }
377  // Finalize
378  for (OrderType i_basis{}; i_basis < n_basis_functions; ++i_basis) {
379  derivatives[i_deriv][i_basis] *= inv_w_fact;
380  }
381  }
382  // Return last value - this version needs an explicit move call to move.
383  return std::move(derivatives[number_of_derivs - 1]);
384 }
