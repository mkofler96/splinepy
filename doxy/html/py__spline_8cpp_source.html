49 void PySpline::NewCore(const py::kwargs& kwargs) {
50 
51  // early exit if this is an incomplete kwargs call
52  if (!kwargs.contains("degrees") || !kwargs.contains("control_points")) {
53  return;
54  }
55 
56  // parse kwargs
57  int* degrees_ptr = nullptr;
58  std::vector<std::vector<double>> knot_vectors;
59  std::vector<std::vector<double>>* knot_vectors_ptr = nullptr;
60  double* control_points_ptr = nullptr;
61  double* weights_ptr = nullptr;
62  int para_dim, dim, /* for checks -> */ ncps, required_ncps{1};
63 
64  // get degrees and set para_dim
65  auto d_array = py::cast<py::array_t<int>>(kwargs["degrees"]);
66  data_["degrees"] = d_array; // keep reference
67  degrees_ptr = static_cast<int*>(d_array.request().ptr);
68  para_dim = d_array.size();
69  knot_vectors.reserve(para_dim);
70 
71  // get cps and set dim. set ncps for checks later
72  auto cp_array = py::cast<py::array_t<double>>(kwargs["control_points"]);
73  data_["control_points"] = cp_array;
74  control_points_ptr = static_cast<double*>(cp_array.request().ptr);
75  dim = cp_array.shape(1);
76  ncps = cp_array.shape(0);
77  if (dim == 0) {
78  splinepy::utils::PrintAndThrowError(
79  "Zero dimensional data points can not constitute a spline.");
80  }
81 
82  // maybe, get knot_vectors
83  if (kwargs.contains("knot_vectors")) {
84  // check list
85  int kv_dim{0};
86  double prev_knot, this_knot;
87  // we don't keep the reference of the knot vectors as we need a specific
88  // type that doesn't duplicate.
89  py::list kvs = kwargs["knot_vectors"];
90 
91  // loop over list of list/arrays
92  for (int i_para{}; i_para < para_dim; i_para++) {
93  // Check degree
94  if (degrees_ptr[i_para] < 0) {
95  splinepy::utils::PrintAndThrowError(
96  "Invalid degree, degrees need to be positive. Detected degree",
97  degrees_ptr[i_para],
98  "along parametric dimension:",
99  i_para);
100  }
101 
102  std::vector<double> knot_vector;
103  // cast to array_t, as python side will try to use tracked array anyways
104  auto kv_array = py::cast<py::array_t<double>>(kvs[i_para]);
105  const int n_knots = kv_array.size();
106  knot_vector.reserve(n_knots);
107 
108  int nknots{0};
109  prev_knot = -1.;
110  for (py::handle k : kvs[i_para]) {
111  this_knot = k.cast<double>();
112 
113  // must be increasing
114  if (prev_knot - this_knot > 0) {
115  splinepy::utils::PrintAndThrowError(prev_knot,
116  this_knot,
117  "Knots of parametric dimension (",
118  kv_dim,
119  ")",
120  "are not in increasing order.");
121  }
122 
123  // lgtm, add!
124  // 0 0 1 1
125  // 0 1 2 3
126  knot_vector.push_back(this_knot);
127 
128  // prepare next
129  prev_knot = this_knot;
130  ++nknots;
131  }
132 
133  // minimal number of knots check
134  int n_minimal_knots = 2 * (degrees_ptr[kv_dim] + 1);
135  if (nknots < n_minimal_knots) {
136  splinepy::utils::PrintAndThrowError(
137  "Not enough knots in parametric dimension (",
138  kv_dim,
139  ").",
140  "At least",
141  n_minimal_knots,
142  "knots are expected, but",
143  nknots,
144  "were given.");
145  }
146 
147  // multiply expected control mesh resolution
148  required_ncps *= nknots - degrees_ptr[kv_dim] - 1;
149 
150  // lgtm, add!
151  knot_vectors.push_back(std::move(knot_vector));
152 
153  // prepare next
154  ++kv_dim;
155  }
156 
157  // check number of control points
158  if (ncps != required_ncps) {
159  splinepy::utils::PrintAndThrowError("Invalid number of control points.",
160  required_ncps,
161  "expected, but",
162  ncps,
163  "were given.");
164  }
165  // last, dim check
166  if (para_dim != kv_dim) {
167  splinepy::utils::PrintAndThrowError(
168  "Dimension mismatch between `degrees` (",
169  para_dim,
170  ") and `knot_vectors` (",
171  kv_dim,
172  ").");
173  }
174 
175  // lgtm, set!
176  knot_vectors_ptr = &knot_vectors;
177  } else {
178  // Perform checks and compute required_ncps
179  required_ncps = 1;
180  for (int i_para{}; i_para < para_dim; i_para++) {
181  if (degrees_ptr[i_para] < 0) {
182  splinepy::utils::PrintAndThrowError(
183  "Invalid degree, degrees need to be positive. Detected degree",
184  degrees_ptr[i_para],
185  "along parametric dimension:",
186  i_para);
187  }
188  required_ncps *= degrees_ptr[i_para] + 1;
189  }
190  }
191 
192  // check number of control points
193  if (ncps != required_ncps) {
194  splinepy::utils::PrintAndThrowError("Invalid number of control points.",
195  required_ncps,
196  "expected, but",
197  ncps,
198  "were given.");
199  }
200 
201  // maybe, get weights
202  if (kwargs.contains("weights")) {
203  auto w_array = py::cast<py::array_t<double>>(kwargs["weights"]);
204  data_["weights"] = w_array;
205  weights_ptr = static_cast<double*>(w_array.request().ptr);
206 
207  // check if size matches with ncps
208  if (w_array.size() != ncps) {
209  splinepy::utils::PrintAndThrowError(
210  "Number of weights (",
211  w_array.size(),
212  ") does not match number of control points (",
213  ncps,
214  ").");
215  }
216  }
217 
218  // new assign
219  c_spline_ =
220  splinepy::splines::SplinepyBase::SplinepyCreate(para_dim,
221  dim,
222  degrees_ptr,
223  knot_vectors_ptr,
224  control_points_ptr,
225  weights_ptr);
226  para_dim_ = c_spline_->SplinepyParaDim();
227  dim_ = c_spline_->SplinepyDim();
228 }

258 py::dict PySpline::CurrentCoreProperties() const {
259  py::dict dict_spline;
260 
261  // prepare property arrays
262  // first, degrees and control_points
263  py::array_t<int> degrees(para_dim_);
264  int* degrees_ptr = static_cast<int*>(degrees.request().ptr);
265  const int ncps = Core()->SplinepyNumberOfControlPoints();
266  py::array_t<double> control_points({ncps, dim_});
267  double* control_points_ptr =
268  static_cast<double*>(control_points.request().ptr);
269 
270  // and maybes
271  std::vector<std::vector<double>> knot_vectors;
272  std::vector<std::vector<double>>* knot_vectors_ptr = nullptr;
273  py::array_t<double> weights;
274  double* weights_ptr = nullptr;
275 
276  if (Core()->SplinepyHasKnotVectors()) {
277  knot_vectors_ptr = &knot_vectors;
278  }
279  if (Core()->SplinepyIsRational()) {
280  weights = py::array_t<double>(ncps);
281  weights_ptr = static_cast<double*>(weights.request().ptr);
282  }
283 
284  Core()->SplinepyCurrentProperties(degrees_ptr,
285  knot_vectors_ptr,
286  control_points_ptr,
287  weights_ptr);
288 
289  // process
290  dict_spline["degrees"] = degrees;
291  dict_spline["control_points"] = control_points;
292 
293  // process maybes
294  if (Core()->SplinepyHasKnotVectors()) {
295  py::list kvs;
296  for (const auto& knot_vector : knot_vectors) {
297  const std::size_t kvsize = knot_vector.size();
298  py::array_t<double> kv(kvsize);
299  double* kv_ptr = static_cast<double*>(kv.request().ptr);
300  std::copy_n(knot_vector.begin(), kvsize, kv_ptr);
301  kvs.append(kv);
302  }
303  dict_spline["knot_vectors"] = kvs;
304  }
305  if (Core()->SplinepyIsRational()) {
306  weights.resize({ncps, 1});
307  dict_spline["weights"] = weights;
308  }
309 
310  return dict_spline;
311 }

376 py::array_t<double> PySpline::Evaluate(py::array_t<double> queries,
377  int nthreads) const {

379 
380  // prepare output

400 py::array_t<double> PySpline::Sample(py::array_t<int> resolutions,
401  int nthreads) const {
402  CheckPyArraySize(resolutions, para_dim_, true);
403 
404  // get sampling bounds
405  std::vector<double> bounds(para_dim_ * 2);
406  double* bounds_ptr = bounds.data();
407  Core()->SplinepyParametricBounds(bounds_ptr);
408  // prepare sampler
409  auto grid =
410  splinepy::utils::GridPoints(para_dim_,
411  bounds_ptr,
412  static_cast<int*>(resolutions.request().ptr));
413  // prepare output
414  const int n_sampled = grid.Size();
415  py::array_t<double> sampled(n_sampled * dim_);
416  double* sampled_ptr = static_cast<double*>(sampled.request().ptr);
417 
418  // wrap evaluate
419  auto sample = [&](const int begin, const int end, int) {
420  std::vector<double> query(para_dim_);
421  double* query_ptr = query.data();
422  for (int i{begin}; i < end; ++i) {
423  grid.IdToGridPoint(i, query_ptr);
424  Core()->SplinepyEvaluate(&query[0], &sampled_ptr[i * dim_]);
425  }
426  };
427 
428  splinepy::utils::NThreadExecution(sample, n_sampled, nthreads);
429 
430  sampled.resize({n_sampled, dim_});
431  return sampled;
432 }

462 py::array_t<double> PySpline::Derivative(py::array_t<double> queries,
463  py::array_t<int> orders,
464  int nthreads) const {
465  // process input
466  CheckPyArrayShape(queries, {-1, para_dim_}, true);
467  const int n_queries = queries.shape(0);
468  int n_orders{};
469  if (CheckPyArraySize(orders, para_dim_, false)) {
470  n_orders = 1;
471  } else if (CheckPyArrayShape(orders, {-1, para_dim_}, false)) {
472  n_orders = orders.shape(0);
473  } else {
474  splinepy::utils::PrintAndThrowError(
475  "Derivative-query-orders must be either have same size as para_dim",
476  "or (n, para_dim) shape.");
477  }
478 
479  // prepare output
480  py::array_t<double> derived({n_queries * n_orders, dim_});
481  double* derived_ptr = static_cast<double*>(derived.request().ptr);
482 
483  // prepare lambda for nthread exe
484  double* queries_ptr = static_cast<double*>(queries.request().ptr);
485  int* orders_ptr = static_cast<int*>(orders.request().ptr);
486 
487  const int out_stride0 = dim_ * n_orders;
488  const int out_stride1 = dim_;
489  auto derive = [&](const int begin, const int end, int) {
490  for (int i{begin}; i < end; ++i) {
491  for (int j{}; j < n_orders; ++j) {
492  Core()->SplinepyDerivative(
493  &queries_ptr[i * para_dim_],
494  &orders_ptr[j * para_dim_],
495  &derived_ptr[i * out_stride0 + j * out_stride1]);
496  }
497  }
498  };
499 
500  splinepy::utils::NThreadExecution(derive, n_queries, nthreads);
501 
502  if (n_orders > 1) {
503  derived.resize({n_queries, n_orders, dim_});
504  }
505  return derived;
506 }

638  if (CheckPyArraySize(orders, para_dim_, false)) {
639  n_orders = 1;
640  } else if (CheckPyArrayShape(orders, {-1, para_dim_}, false)) {
641  n_orders = orders.shape(0);
642  } else {
643  splinepy::utils::PrintAndThrowError(
644  "Derivative-query-orders must be either have same size as para_dim",
645  "or (n, para_dim) shape.");
646  }
647 
648  // prepare results
649  const int n_support = Core()->SplinepyNumberOfSupports();
650  py::array_t<double> basis_der({n_queries * n_orders, n_support});
651  py::array_t<int> support({n_queries, n_support});
652 
653  // prepare_lambda for nthread exe
654  double* queries_ptr = static_cast<double*>(queries.request().ptr);
655  int* orders_ptr = static_cast<int*>(orders.request().ptr);
656  double* basis_der_ptr = static_cast<double*>(basis_der.request().ptr);
657  int* support_ptr = static_cast<int*>(support.request().ptr);
658 
659  const int out_stride0 = n_support * n_orders;
660  const int out_stride1 = n_support;
661 
662  auto basis_der_support = [&](const int begin, const int end, int) {
663  for (int i{begin}; i < end; ++i) {
664  // first fill ders
665  for (int j{}; j < n_orders; ++j) {
666  Core()->SplinepyBasisDerivative(
667  &queries_ptr[i * para_dim_],
668  &orders_ptr[j * para_dim_],
669  &basis_der_ptr[i * out_stride0 + j * out_stride1]);
670  }
671  // then supports
672  Core()->SplinepySupport(&queries_ptr[i * para_dim_],
673  &support_ptr[i * n_support]);
674  }
675  };
676 
677  splinepy::utils::NThreadExecution(basis_der_support, n_queries, nthreads);
678 
679  if (n_orders > 1) {
680  basis_der.resize({n_queries, n_orders, n_support});
681  }
682 
683  return py::make_tuple(basis_der, support);
684 }

686 py::tuple
687 PySpline::Proximities(py::array_t<double> queries,
688  py::array_t<int> initial_guess_sample_resolutions,
689  const double tolerance,
690  const int max_iterations,
691  const bool aggresive_search_bounds,
692  const int nthreads) {
693  CheckPyArrayShape(queries, {-1, dim_}, true);
694  CheckPyArraySize(initial_guess_sample_resolutions, para_dim_);
695 
696  const int n_queries = queries.shape(0);
697  const int pd = para_dim_ * dim_;
698  const int ppd = para_dim_ * pd;
699 
700  // prepare results
701  py::array_t<double> para_coord({n_queries, para_dim_});
702  py::array_t<double> phys_coord({n_queries, dim_});
703  py::array_t<double> phys_diff({n_queries, dim_});
704  py::array_t<double> distance({n_queries, 1});
705  py::array_t<double> convergence_norm({n_queries, 1});
706  py::array_t<double> first_derivatives({n_queries, para_dim_, dim_});
707  py::array_t<double> second_derivatives(
708  {n_queries, para_dim_, para_dim_, dim_});
709 
710  // prepare lambda for nthread exe
711  double* queries_ptr = static_cast<double*>(queries.request().ptr);
712  double* para_coord_ptr = static_cast<double*>(para_coord.request().ptr);
713  double* phys_coord_ptr = static_cast<double*>(phys_coord.request().ptr);
714  double* phys_diff_ptr = static_cast<double*>(phys_diff.request().ptr);
715  double* distance_ptr = static_cast<double*>(distance.request().ptr);
716  double* convergence_norm_ptr =
717  static_cast<double*>(convergence_norm.request().ptr);
718  double* first_derivatives_ptr =
719  static_cast<double*>(first_derivatives.request().ptr);
720  double* second_derivatives_ptr =
721  static_cast<double*>(second_derivatives.request().ptr);
722  auto proximities = [&](const int begin, const int end, int) {
723  for (int i{begin}; i < end; ++i) {
724  Core()->SplinepyVerboseProximity(&queries_ptr[i * dim_],
725  tolerance,
726  max_iterations,
727  aggresive_search_bounds,
728  &para_coord_ptr[i * para_dim_],
729  &phys_coord_ptr[i * dim_],
730  &phys_diff_ptr[i * dim_],
731  distance_ptr[i],
732  convergence_norm_ptr[i],
733  &first_derivatives_ptr[i * pd],
734  &second_derivatives_ptr[i * ppd]);
735  }
736  };
737 
738  // make sure kdtree is planted before query.
739  //
740  // there are two cases, where tree will not be built:
741  // 1. same resolution as last proximity call - core spline will check
742  // 2. any negative entry in resolutions - checked here
743  //
744  // there is one cases, where runtime_error will be thrown:
745  // 1. entry smaller than 2 - checked here
746  //
747  // allow us to use abbreviation here.
748  int* igsr_ptr =
749  static_cast<int*>(initial_guess_sample_resolutions.request().ptr);
750  bool plant_kdtree = true;
751  for (int i{}; i < para_dim_; ++i) {
752  const int& res = igsr_ptr[i];
753  if (res < 0) {
754  plant_kdtree = false;
755  break;
756  }
757  if (res < 2) {
758  splinepy::utils::PrintAndThrowError(
759  "positive entries for initial_guess_sample_resolutions",
760  "shouldn't be smaller than 2.",
761  "Entry number: [",
762  i,
763  "] is",
764  res);
765  }
766  }
767  // yes, we could've built an input and called the function directly,
768  // but, we will stick with calling interface functions
769  if (plant_kdtree) {
770  Core()->SplinepyPlantNewKdTreeForProximity(igsr_ptr, nthreads);
771  }
772 
773  splinepy::utils::NThreadExecution(proximities, n_queries, nthreads);
774 
775  return py::make_tuple(para_coord,
776  phys_coord,
777  phys_diff,
778  distance,
779  convergence_norm,
780  first_derivatives,
781  second_derivatives);
782 }
