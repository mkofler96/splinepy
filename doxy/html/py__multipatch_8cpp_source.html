104 IntVector ArgSort(const std::vector<T>& v) {
105  IntVector idx(v.size());
106  std::iota(idx.begin(), idx.end(), 0);
107  std::stable_sort(idx.begin(), idx.end(), [&v](int const& i1, int const& i2) {

113 void RaiseMismatch(const CoreSplineVector& splist,
114  const std::string name,
115  const int para_dim,
116  const int dim,
117  const IntVector& degrees,
118  const IntVector& control_mesh_resolutions,
119  const int nthreads) {
120  // for verbose output
121  std::unordered_map<std::string, IntVectorVector> mismatches{};
122 
123  // check flags
124  bool check_name{}, check_para_dim{}, check_dim{}, check_degrees{},
125  check_control_mesh_resolutions{};
126  // input vector sizes
127  const int d_size = degrees.size();
128  const int cmr_size = control_mesh_resolutions.size();
129  // true para_dim to use as reference -> will be updated / compared
130  int ref_para_dim{para_dim};
131 
132  // parse input and allocate vector for mismatch book keeping
133  if (name.size() > 0) {
134  check_name = true;
135  mismatches["name"].resize(nthreads);
136  }
137  if (dim > 0) {
138  check_dim = true;
139  mismatches["dim"].resize(nthreads);
140  }
141  if (d_size > 0) {
142  check_degrees = true;
143  ref_para_dim = d_size;
144  mismatches["degrees"].resize(nthreads);
145  }
146  if (cmr_size > 0) {
147  check_control_mesh_resolutions = true;
148  mismatches["control_mesh_resolutions"].resize(nthreads);
149  }
150  if (para_dim > 0 || check_degrees || check_control_mesh_resolutions) {
151  check_para_dim = true;
152 
153  // sanity check
154  for (const auto& candidate : {para_dim, d_size, cmr_size}) {
155  if (candidate != ref_para_dim && candidate > 0) {
156  splinepy::utils::PrintAndThrowError(
157  "Mismatch in given para_dim (",
158  para_dim,
159  "), size of degrees, (",
160  d_size,
161  "), size of control_mesh_resolutions (",
162  cmr_size,
163  ").");
164  }
165  }
166  mismatches["para_dim"].resize(nthreads);
167  }
168 
169  // lambda for nthread comparison
170  auto check_mismatch_step = [&](int, int, const int i_thread) {
171  // alloc vectors in case we need to compare
172  IntVector spline_degree(ref_para_dim), spline_cmr(ref_para_dim);
173  int total_ = static_cast<int>(splist.size());
174  for (int i{i_thread}; i < total_; i += nthreads) {
175  // get spline to check
176  const auto& spline = splist[i];
177 
178  // skip null splines
179  if (spline->SplinepyIsNull()) {
180  continue;
181  }
182 
183  // default value for para_dim match
184  bool para_dim_matches{true};
185 
186  // check
187  if (check_name && (name != spline->SplinepySplineName())) {
188  mismatches["name"][i_thread].push_back(i);
189  }
190  if (check_para_dim && (ref_para_dim != spline->SplinepyParaDim())) {
191  mismatches["para_dim"][i_thread].push_back(i);
192  para_dim_matches = false;
193  }
194  if (check_dim && (dim != spline->SplinepyDim())) {
195  mismatches["dim"][i_thread].push_back(i);
196  }
197  // check properties that are relevant iff para_dim matches
198  if (para_dim_matches) {
199  if (check_degrees) {
200  spline->SplinepyCurrentProperties(spline_degree.data(),
201  nullptr,
202  nullptr,
203  nullptr);
204  if (spline_degree != degrees) {
205  mismatches["degrees"][i_thread].push_back(i);
206  }
207  }
208  if (check_control_mesh_resolutions) {
209  spline->SplinepyControlMeshResolutions(spline_cmr.data());
210  if (spline_cmr != control_mesh_resolutions) {
211  mismatches["control_mesh_resolutions"][i_thread].push_back(i);
212  }
213  }
214  }
215  }
216  };
217 
218  splinepy::utils::NThreadExecution(check_mismatch_step,
219  static_cast<int>(splist.size()),
220  nthreads);
221 
222  // prepare output or maybe exit
223  std::unordered_map<std::string, IntVector> concat_mismatches{};
224  bool raise{false};
225 
226  for (const auto& [key, mismatch_per_threads] : mismatches) {
227  auto& concat_vec = concat_mismatches[key];
228  for (const auto& mismatch : mismatch_per_threads) {
229  const auto m_size = mismatch.size();
230  if (m_size != 0) {
231  concat_vec.insert(concat_vec.end(), mismatch.begin(), mismatch.end());
232  }
233  }
234  if (concat_vec.size() != 0) {
235  raise = true;
236  }
237  };
238 
239  // everything matches
240  if (!raise) {
241  return;
242  }
243 
244  // form mismatch info
245  std::string mismatch_info{};
246  for (const auto& [key, concat_mismatch] : concat_mismatches) {
247  mismatch_info += "\n[" + key + "] : ";
248  for (const auto& ids : concat_mismatch) {
249  mismatch_info += std::to_string(ids) + ", ";
250  }
251  }
252 
253  splinepy::utils::PrintAndThrowError("Found mismatches.", mismatch_info);
254 }

396 py::array_t<int>
397 FindConnectivityFromCenters(const py::array_t<double>& face_center_vertices,
398  const int parametric_dimension,
399  const py::array_t<double>& metric,
400  const double tolerance) {
401 
402  // -- Auxiliary data --
403  const int number_of_element_faces = parametric_dimension * 2;
404  const int number_of_patches =
405  face_center_vertices.shape(0) / number_of_element_faces;
406  const int physical_dimension = face_center_vertices.shape(1);
407  const int number_of_center_vertices{
408  static_cast<int>(face_center_vertices.shape(0))};
409  const double tolerance_squared{tolerance * tolerance};
410  const double* metric_ptr = static_cast<double*>(metric.request().ptr);
411  const double* face_center_vertices_ptr =
412  static_cast<double*>(face_center_vertices.request().ptr);
413 
414  // Consistency check
415  if (!(face_center_vertices.shape(0) % number_of_element_faces == 0)) {
416  splinepy::utils::PrintAndThrowError(
417  "Number of corner vertices invalid. Must be a multiple of the number "
418  "of vertices per patch");
419  }
420  if (!(metric.size() == face_center_vertices.shape(1))) {
421  splinepy::utils::PrintAndThrowError(
422  "Incompatible size for metric. Must match physical dimension of face "
423  "center vertices");
424  }
425  if (!(number_of_center_vertices % number_of_element_faces == 0)) {
426  splinepy::utils::PrintAndThrowError(
427  "Inconsistent number of Center vertices. Must be divisible by "
428  "parametric_dimension*2");
429  }
430 
431  // Init return type
432  py::array_t<int> connectivity(number_of_patches * number_of_element_faces);
433  int* connectivity_ptr = static_cast<int*>(connectivity.request().ptr);
434  // Init connectivity and metric value
435  // (-1 : boundary, -2 : untouched)
436  std::fill(connectivity_ptr,
437  connectivity_ptr + number_of_patches * number_of_element_faces,
438  -2);
439 
440  // Assure Metric is normed and non-zero
441  const std::vector<double>& normed_metric = [&metric_ptr,
442  &physical_dimension]() {
443  double euclidian_norm = 0.;
444  for (int i_phys{}; i_phys < physical_dimension; i_phys++) {
445  euclidian_norm += metric_ptr[i_phys] * metric_ptr[i_phys];
446  }
447  std::vector<double> metric(physical_dimension, 1.);
448  euclidian_norm = std::sqrt(euclidian_norm);
449  if (euclidian_norm < 1e-20) {
450  return metric;
451  } else {
452  const double inv_euclidian_norm = 1 / euclidian_norm;
453  for (int i_phys{}; i_phys < physical_dimension; i_phys++) {
454  metric[i_phys] = metric_ptr[i_phys] * inv_euclidian_norm;
455  }
456  }
457  return metric;
458  }();
459 
460  // Auxiliary function to determine distance between two points in face_center
461  auto squared_euclidian_distance =
462  [&face_center_vertices_ptr,
463  &physical_dimension](const int& i_start, const int& i_end) -> double {
464  double squared_euclidian_distance_{};
465  for (int i_phys{}; i_phys < physical_dimension; i_phys++) {
466  const double distance_c =
467  face_center_vertices_ptr[i_end * physical_dimension + i_phys]
468  - face_center_vertices_ptr[i_start * physical_dimension + i_phys];
469  squared_euclidian_distance_ += distance_c * distance_c;
470  }
471  return squared_euclidian_distance_;
472  };
473 
474  std::vector<double> scalar_metric{};
475  scalar_metric.reserve(number_of_center_vertices);
476 
477  // Check Metric Dimension and Vector Size
478  for (int i_vertex{}; i_vertex < number_of_center_vertices; i_vertex++) {
479  double metric_v = normed_metric[0]
480  * face_center_vertices_ptr[i_vertex * physical_dimension];
481  for (int j_phys{1}; j_phys < physical_dimension; j_phys++) {
482  metric_v =
483  normed_metric[j_phys]
484  * face_center_vertices_ptr[i_vertex * physical_dimension + j_phys];
485  }
486  scalar_metric.push_back(metric_v);
487  }
488 
489  // Sort Metric Vector
490  const auto metric_order_indices = ArgSort(scalar_metric);
491 
492  // Loop over points
493  for (int lower_limit = 0; lower_limit < number_of_center_vertices - 1;
494  lower_limit++) {
495  // Loop over all points regardless of whether they have been touched or not,
496  // and then check the validity of the connection Point already processed
497  bool found_duplicate = false;
498  // Now check allowed range for duplicates
499  int upper_limit = lower_limit + 1;
500  while (upper_limit < number_of_center_vertices
501  && (scalar_metric[metric_order_indices[upper_limit]]
502  - scalar_metric[metric_order_indices[lower_limit]])
503  < tolerance) {
504  // Check if the two points are duplicates
505  found_duplicate =
506  squared_euclidian_distance(metric_order_indices[lower_limit],
507  metric_order_indices[upper_limit])
508  < tolerance_squared;
509  if (found_duplicate) {
510  break;
511  } else {
512  upper_limit++;
513  }
514  }
515 
516  // Now we have to check if the connection is valid
517  // 1. If another connection is found, that means, that the point it connects
518  // to has a higher index in the metric tensor. If the current point does
519  // already have a neighbor, that means that more than one point connect
520  // -> Error
521  if (found_duplicate) {
522  if (connectivity_ptr[metric_order_indices[lower_limit]] != -2) {
523  splinepy::utils::PrintAndThrowError(
524  "Found conflicting interceptions, where more than two points are "
525  "in the same position. Expected -2 got",
526  connectivity_ptr[metric_order_indices[lower_limit]]);
527  }
528 
529  // If both tests passed, update connectivity
530  connectivity_ptr[metric_order_indices[lower_limit]] =
531  static_cast<int>(metric_order_indices[upper_limit]);
532  connectivity_ptr[metric_order_indices[upper_limit]] =
533  static_cast<int>(metric_order_indices[lower_limit]);
534  } else {
535  // set Boundary-ID
536  if (connectivity_ptr[metric_order_indices[lower_limit]] == -2) {
537  connectivity_ptr[metric_order_indices[lower_limit]] = -1;
538  }
539  }
540  }
541 
542  // Treat last remaining point in scalar metric vector
543  if (connectivity_ptr[metric_order_indices[number_of_center_vertices - 1]]
544  == -2) {
545  connectivity_ptr[metric_order_indices[number_of_center_vertices - 1]] = -1;
546  }
547 
548  // Resize buffer
549  connectivity.resize({number_of_patches, number_of_element_faces});
550  return connectivity;
551 }

553 py::array_t<int>
554 InterfacesFromBoundaryCenters(const py::array_t<double>& py_center_vertices,
555  const double& tolerance,
556  const int& parametric_dimension) {
557  // Determine Metric
558  const int physical_dimension = py_center_vertices.shape(1);
559  const int number_of_corner_points = py_center_vertices.shape(0);
560  py::array_t<double> maxvertex(physical_dimension),
561  minvertex{physical_dimension};
562 
563  // Get pointers
564  double* maxvertex_ptr = static_cast<double*>(maxvertex.request().ptr);
565  double* minvertex_ptr = static_cast<double*>(minvertex.request().ptr);
566  const double* py_center_vertices_ptr =
567  static_cast<double*>(py_center_vertices.request().ptr);
568 
569  // Calculate limits of center vertices
570  for (int i_dim{}; i_dim < physical_dimension; i_dim++) {
571  minvertex_ptr[i_dim] = py_center_vertices_ptr[i_dim];
572  maxvertex_ptr[i_dim] = py_center_vertices_ptr[i_dim];
573  }
574  for (int i_c{1}; i_c < number_of_corner_points; i_c++) {
575  for (int i_dim{}; i_dim < physical_dimension; i_dim++) {
576  maxvertex_ptr[i_dim] =
577  std::max(py_center_vertices_ptr[i_c * physical_dimension + i_dim],
578  maxvertex_ptr[i_dim]);
579  minvertex_ptr[i_dim] =
580  std::min(py_center_vertices_ptr[i_c * physical_dimension + i_dim],
581  minvertex_ptr[i_dim]);
582  }
583  }
584 
585  // Determine Interfaces
586  return FindConnectivityFromCenters(py_center_vertices,
587  parametric_dimension,
588  maxvertex - minvertex,
589  tolerance);
590 }

625  mappings_ptr[boundary_start_p_dim] = boundary_end_p_dim;
626  // Note: Here might be a discrepancy with gismo's orientation, and it needs to
627  // be checked in the future. I am awaiting a response from gismo developers,
628  // it is poosible the orientation of the interface edge might be flipped
629  // (bugfix: negate the following expression)
630  orientations_ptr[boundary_start_p_dim] =
631  (boundary_start_orientation ^ boundary_end_orientation) ? 1 : 0;
632 
634  // Calculate Parametric bounds
635  std::vector<double> bounds_start(para_dim_ * 2);
636  pyspline_start->SplinepyParametricBounds(bounds_start.data());
637  std::vector<double> bounds_end(para_dim_ * 2);
638  pyspline_end->SplinepyParametricBounds(bounds_end.data());
639  // Determine face center position in parametric space
640  std::vector<double> boundary_center_start(para_dim_),
641  boundary_center_end(para_dim_);
642 
643  for (int i{}; i < para_dim_; i++) {
644  if (i == boundary_start_p_dim) {
645  boundary_center_start[i] = boundary_start_orientation
646  ? bounds_start[i]
647  : bounds_start[i + para_dim_];
648  } else {
649  boundary_center_start[i] =
650  .5 * (bounds_start[i] + bounds_start[i + para_dim_]);
651  }
652  if (i == boundary_end_p_dim) {
653  boundary_center_end[i] =
654  boundary_end_orientation ? bounds_end[i] : bounds_end[i + para_dim_];
655  } else {
656  boundary_center_end[i] = .5 * (bounds_end[i] + bounds_end[i + para_dim_]);
657  }
658  }
659 
660  // Calculate Jacobians
661  std::vector<double> jacobian_start(para_dim_ * dim_),
662  jacobian_end(para_dim_ * dim_);
663  pyspline_start->SplinepyJacobian(boundary_center_start.data(),
664  jacobian_start.data());
665  pyspline_end->SplinepyJacobian(boundary_center_end.data(),
666  jacobian_end.data());
667 
668  // Check the angle between the jacobian entries
669  for (int i_pd{}; i_pd < para_dim_; i_pd++) {
670  if (i_pd == boundary_start_p_dim) {
671  continue;
672  }
673  double norm_s{};
674  for (int k{}; k < dim_; k++) {
675  // [i_query * pdim * dim + i_paradim * dim + i_dim]
676  norm_s += jacobian_start[i_pd + k * para_dim_]
677  * jacobian_start[i_pd + k * para_dim_];
678  }
679  for (int j{}; j < para_dim_; j++) {
680  double norm_e{}, dot_p{};
681  for (int k{}; k < dim_; k++) {
682  dot_p += jacobian_start[i_pd + k * para_dim_]
683  * jacobian_end[j + k * para_dim_];
684  norm_e +=
685  jacobian_end[j + k * para_dim_] * jacobian_end[j + k * para_dim_];
686  }
687 
688  // Check angle
689  const double cos_angle = abs(dot_p / std::sqrt(norm_s * norm_e));
690  if (cos_angle > (1. - tolerance)) {
691  mappings_ptr[i_pd] = j;
692  orientations_ptr[i_pd] = (dot_p > 0) ? 1 : 0;
693  break;
694  }
695  }
696  }

848 int AddBoundariesFromContinuity(const py::list& boundary_splines,
849  const py::array_t<int>& boundary_interfaces,
850  py::array_t<int>& global_interfaces,
851  const double& tolerance,
852  const int& n_threads) {
853  // Check input data
854  if (static_cast<int>(py::len(boundary_splines))
855  != boundary_interfaces.shape(0)) {
856  splinepy::utils::PrintAndThrowError(
857  "Number of splines in list (",
858  py::len(boundary_splines),
859  ") and number of elements in connectivity (",
860  boundary_interfaces.shape(0),
861  ") does not match.");
862  }
863 
864  // Provide auxiliary values
865  const auto cpp_spline_list = ToCoreSplineVector(boundary_splines);
866 
867  const int n_boundary_patches{static_cast<int>(boundary_interfaces.shape(0))};
868  const int n_faces_per_boundary_patch{
869  static_cast<int>(boundary_interfaces.shape(1))};
870  const int para_dim_{n_faces_per_boundary_patch / 2};
871  const int dim_ = cpp_spline_list[0]->SplinepyDim();
872  const int* boundary_interfaces_ptr =
873  static_cast<int*>(boundary_interfaces.request().ptr);
874  int* global_interfaces_ptr =
875  static_cast<int*>(global_interfaces.request().ptr);
876 
877  // This function is currently only implemented for 2D and 3D applications. For
878  // higher dimensions, please revise the implementation of the generalized
879  // cross product as can be found here:
880  // https://math.stackexchange.com/questions/752774/generalized-cross-product
881  if ((dim_ < 1) || (dim_ > 3)) {
882  splinepy::utils::PrintAndThrowError(
883  "G1-based boundary computation is only supported for 2D and 3D "
884  "multipatches. If this is a feature you require, please consider "
885  "writing an issue on our github");
886  }
887 
888  // Auxiliary Lambdas to keep code clean
889  // Check if to normal vectors are g1 (tol > cos(phi))
890  auto areG1 = [&tolerance, &dim_](const std::vector<double>& vec0,
891  const std::vector<double>& vec1) -> bool {
892  // Checks in Debug
893  assert(static_cast<int>(vec0.size()) == dim_);
894  assert(static_cast<int>(vec1.size()) == dim_);
895 
896  // Start actual computation
897  double norm0{}, norm1{}, dot_p{};
898  for (int i{}; i < dim_; i++) {
899  norm0 += vec0[i] * vec0[i];
900  norm1 += vec1[i] * vec1[i];
901  dot_p += vec0[i] * vec1[i];
902  }
903  return (tolerance > abs(1 - abs(dot_p) / std::sqrt(norm0 * norm1)));
904  };
905 
906  // Tangential Vector on boundary based on its derivative
907  auto normal_vector = [&cpp_spline_list, &para_dim_, &dim_](
908  const int& patch_id,
909  const int& face_id) -> std::vector<double> {
910  // init return value (are default initialized to 0)
911  std::vector<double> para_coord(para_dim_), bounds(2 * para_dim_),
912  normal_vector(dim_), jacobian(dim_ * para_dim_);
913 
914  // Auxiliary values
915  const int axis_dim = face_id / 2;
916  const int is_in_front = face_id % 2;
917 
918  // Parametric Bounds
919  const auto& spline = cpp_spline_list[patch_id];
920  spline->SplinepyParametricBounds(bounds.data());
921 
922  for (int i{}; i < para_dim_; i++) {
923  if (i == axis_dim) {
924  para_coord[i] = bounds[i + is_in_front * para_dim_];
925  } else {
926  para_coord[i] = .5 * (bounds[i + para_dim_] + bounds[i]);
927  }
928  }
929 
930  // Compute Jacobian
931  spline->SplinepyJacobian(para_coord.data(), jacobian.data());
932 
933  // Compute Cross-Product
934  if (dim_ == 2) {
935  normal_vector[0] = -jacobian[1];
936  normal_vector[1] = jacobian[0];
937  } else if (dim_ == 3) {
938  normal_vector[0] = jacobian[2] * jacobian[5] - jacobian[4] * jacobian[3];
939  normal_vector[1] = jacobian[4] * jacobian[1] - jacobian[0] * jacobian[5];
940  normal_vector[2] = jacobian[0] * jacobian[3] - jacobian[2] * jacobian[1];
941  } else {
942  // This should never happen!
943  splinepy::utils::PrintAndThrowError(
944  "G1-based boundary computation is only supported for 2D and 3D "
945  "multipatches. If this is a feature you require, please consider "
946  "writing an issue on our github");
947  }
948  return normal_vector;
949  };
950 
951  // Start Computations ------------------------------------------------ //
952  // while the actual propagation needs to be performed in serial, the
953  // precomputation of interface tolerances can be performed in parallel
954  std::vector<bool> faces_are_g1(n_faces_per_boundary_patch
955  * n_boundary_patches);
956  auto precompute_tolerances = [&](const int start, const int end, int) {
957  // Loop over relevant faces
958  for (int i{start}; i < end; i++) {
959  // Loop over faces
960  for (int j{}; j < n_faces_per_boundary_patch; j++) {
961  // Interface refers to the global face id (see interface documentation)
962  const int& connected_face_id =
963  boundary_interfaces_ptr[i * n_faces_per_boundary_patch + j];
964 
965  // Retrieve patch and face ID of neighbor
966  const int adjacent_element_id =
967  connected_face_id / n_faces_per_boundary_patch;
968  const int adjacent_face_id =
969  connected_face_id % n_faces_per_boundary_patch;
970 
971  // only compute if the adjacent neighbor has higher id to prevent double
972  // the work
973  if (adjacent_element_id < i) {
974  continue;
975  }
976  // Get tangential vector of current patch
977  const std::vector<double> vec0 = normal_vector(i, j);
978 
979  // Get corresponding tangential vector of neighbor patch
980  const std::vector<double> vec1 =
981  normal_vector(adjacent_element_id, adjacent_face_id);
982 
983  // Check tolerance
984  const bool is_g1 = areG1(vec0, vec1);
985  faces_are_g1[i * n_faces_per_boundary_patch + j] = is_g1;
986  faces_are_g1[connected_face_id] = is_g1;
987  }
988  }
989  };
990 
991  // Execute in parallel
992  splinepy::utils::NThreadExecution(precompute_tolerances,
993  n_boundary_patches,
994  n_threads);
995 
996  // std::vector can use less memory for bools
997  std::vector<bool> is_assigned(n_boundary_patches); // defaults false
998  std::vector<int> new_boundary_id(n_boundary_patches);
999  std::vector<int> queued_splines{};
1000 
1001  // Start Assignment
1002  // Loop over all patches
1003  int current_max_id{1};
1004  for (int i{}; i < n_boundary_patches; i++) {
1005  if (is_assigned[i]) {
1006  continue;
1007  }
1008  new_boundary_id[i] = current_max_id;
1009  is_assigned[i] = true;
1010  queued_splines.push_back(i);
1011 
1012  // Start propagation
1013  while (!queued_splines.empty()) {
1014  const int current_id = queued_splines.back();
1015  queued_splines.pop_back();
1016  for (int i_face{}; i_face < n_faces_per_boundary_patch; i_face++) {
1017  const int combined_index =
1018  current_id * n_faces_per_boundary_patch + i_face;
1019  // Is the neighborface G1
1020  if (faces_are_g1[combined_index]) {
1021  // Adjacent global face id
1022  const int& adjacent_global_face_id =
1023  boundary_interfaces_ptr[combined_index];
1024  const int adjacent_patch_id =
1025  adjacent_global_face_id / n_faces_per_boundary_patch;
1026 
1027  // Check if the adjacent patch is already assigned
1028  if (is_assigned[adjacent_patch_id]) {
1029  continue;
1030  } else {
1031  // Assign a BID and continue
1032  new_boundary_id[adjacent_patch_id] = current_max_id;
1033  is_assigned[adjacent_patch_id] = true;
1034  queued_splines.push_back(adjacent_patch_id);
1035  }
1036  }
1037  }
1038  }
1039 
1040  // End propagation and increase id
1041  current_max_id++;
1042  }
1043 
1044  // Assign the new boundary ids to the old interface-vector
1045  const int& n_interfaces = global_interfaces.size();
1046  int counter{};
1047  for (int i{}; i < n_interfaces; i++) {
1048  if (global_interfaces_ptr[i] < 0) {
1049  global_interfaces_ptr[i] = -new_boundary_id[counter];
1050  counter++;
1051  }
1052  }
1053  if (counter != n_boundary_patches) {
1054  splinepy::utils::PrintAndThrowError(
1055  counter,
1056  " new boundary ids were assigned, however ",
1057  n_boundary_patches,
1058  " were expected, which means information was lost. Abort mission");
1059  }
1060 
1061  return current_max_id;
1062 }
