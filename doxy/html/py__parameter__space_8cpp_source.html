38 void init_parameter_space(py::module_& m) {
39  using PSpace = bsplinelib::parameter_spaces::ParameterSpaceBase;
40  using KV = bsplinelib::parameter_spaces::KnotVector;
41  using KVPtr = std::shared_ptr<KV>;
42 
43  // implementations adapted from pybind11/stl_bind.h
45  auto wrap_id = [](int i, int n) -> int {
46  if (i < 0) {
47  i += n;
48  }
49  if (i < 0 || i >= n) {
50  throw py::index_error();
51  }
52  return i;
53  };

55  auto to_list = [](const PSpace& p) -> py::list {
56  // if this were to be a list, this would mean a shallow copy.
57  // however, within splinepy's implementation, copy() means deepcopy.
58  // so, deepcopy it is.
59  py::list kvs(p.ParaDim());
60  for (int i{}; i < p.ParaDim(); ++i) {
61  KVPtr kv = p.GetKnotVector(i);
62  py::array_t<double> out_kv(kv->GetSize());
63  std::copy_n(kv->GetKnots().data(),
64  kv->GetKnots().size(),
65  static_cast<double*>(out_kv.request().ptr));
66  kvs[i] = out_kv;
67  }
68  return kvs;
69  };

177  py::list unique_knots;
178 
179  for (int i{}; i < p.ParaDim(); ++i) {
180  const int degree = p.GetDegree(i);
181  const auto& kv = p.GetKnotVector(i);
182  const auto& knots = kv->GetKnots();
183 
184  // create array
185  // this uses `new` to allocate memory and we can use this with
186  // py::capsule to avoid copy or resize at the end.
187  // See: github.com/pybind/pybind11/issues/1042#issuecomment-325941022
188  splinepy::utils::Array<double> u_knots(static_cast<int>(knots.size())
189  - (2 * degree));
190 
191  // create a capsule to release data once it's done
192  auto capsule = py::capsule(u_knots.data(), [](void* data) {
193  delete reinterpret_cast<double*>(data);
194  });
195 
196  // unlike KnotVector::UniqueKnots, this checks within
197  // valid regions.
198  int n_uk{1};
199  std::unique_copy(
200  knots.begin() + degree,
201  knots.end() - degree,
202  u_knots.data(),
203  [&n_uk](const double& lhs_knot, const double& rhs_knot) {
204  if (std::abs(lhs_knot - rhs_knot)
205  < bsplinelib::parameter_spaces::kEpsilon) {
206  return true;
207  } else {
208  ++n_uk;
209  return false;
210  };
211  });
212 
213  unique_knots.append(
214  py::array_t<double>(n_uk, u_knots.data(), capsule));
215 
216  // mark ownership to false.
217  // we can do it in append(), but if that line fails, then it'd
218  // be a memory leak.
219  u_knots.TransferOwnership();
220  }
221 
222  return unique_knots;
223  });
224 }
