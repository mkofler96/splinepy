224  template<typename... Ts>
225  constexpr void SetShape(const IndexType& shape0, const Ts&... shape) {
226  static_assert(sizeof...(Ts) == static_cast<std::size_t>(dim - 1),
227  "shape should match the dim");
228

235

243  IndexType sub_total{1};
244  typename StridesType_::reverse_iterator r_stride_iter = strides_.rbegin();
245  for (; r_stride_iter != strides_.rend(); ++r_stride_iter) {
246  // accumulate
247  sub_total *= *r_stride_iter;
248  // assign
249  *r_stride_iter = sub_total;

485  static_assert(dim == 2, "AAt is only applicable for dim=2 array.");
486 
487  const auto& height = shape_[0];
488  const auto& width = shape_[1];

490  for (int i{}; i < height; ++i) {
491  // compute upper triangle
492  for (int j{i}; j < height; ++j) {
493

497  ij = 0.0;

499  const DataType* i_ptr = &operator()(i, 0);

604 template<typename InputArray, typename OutIndexType>
605 void UniqueIndicesAndMultiplicities(
606  const InputArray& input,
607  const Array<OutIndexType, 1, OutIndexType>& arg_sorted,
608  Array<OutIndexType, 1, OutIndexType>& unique,
609  Array<OutIndexType, 1, OutIndexType>& multiplicity) {
610  static_assert(std::is_integral_v<OutIndexType>,
611  "OutIndexType should be an integral type");
612  using InputType = typename InputArray::value_type;
613 
614  // copy argsort to use it for unique search
615  unique.Reallocate(arg_sorted.size());
616  unique = arg_sorted;
617 
618  // comparison for unique check - need to look at input element of given index
619  auto is_same = [&input](const InputType& a, const InputType& b) {
620  if constexpr (std::is_floating_point_v<InputType>) {
621  return std::abs(input[a] - input[b])
622  < std::numeric_limits<InputType>::epsilon();
623  } else {
624  return input[a] == input[b];
625  }
626  };
627 
628  // find unique and reshape
629  auto last = std::unique(unique.begin(), unique.end(), is_same);
630  unique.SetShape(std::distance(unique.begin(), last));
631 
632  // now loop and fill return values
633  multiplicity.Reallocate(unique.size());
634  OutIndexType* unique_ptr = unique.begin();
635 
636  // iterate unique and count multiplicity
637  // we will check if the next indices are the same. if so, we will stop
638  // counting multiplicity
639  const OutIndexType* arg_ptr = arg_sorted.begin();
640  for (OutIndexType i{}; i < unique.size() - 1; ++i) {
641  const OutIndexType u_next = unique[i + 1];
642  OutIndexType& m = multiplicity[i];
643  m = 1; // at start, we always have the same value
644  while (*(++arg_ptr) != u_next) {
645  ++m; // increase counter if the next argument matches
646  }
647  }
648 
649  // last value of multiplicity is distance between
650  OutIndexType* last_multiplicity = multiplicity.end() - 1;
651  *last_multiplicity = std::distance(arg_ptr, arg_sorted.cend());
652 }

707  template<typename IterableB, typename IterableX>
708  constexpr void Solve(IterableB& b, IterableX& x) {
709 
710  // get length of the array/matrix
711  const IndexType& len = array_.Shape()[0];
712 
713  // array_ should be a square matrix
714  assert(len == array_.Shape()[1]);
715  // len should match the size of b and x
716  assert(len == b.size());
717  assert(len == x.size());
718 
719 #ifndef NDEBUG
720  // Create a copy of the linear system for debugging
721  const auto array_copy = array_;
722  const auto b_copy = b;
723 #endif
724 
725  // partial pivoting and forward reduction
726  // since we reorder indices only, we append *_r to reordered ids
727  for (IndexType i{0}; i < len; i++) {
728  // sneak in x initialization here.
729  // it doesn't matter if this is i or i_r
730  x[i] = 0.;
731 
732  // ignore clipped entries
733  // if (skipmask[i] != 0)
734  // continue;
735 
736  /* swap */
737  // go through the rows and compare magnitude and find max row at col i.
738  // we use reordered index here to retrieve rows, but keep the
739  // true index for everything else
740  const IndexType& i_r = row_order_[i];
741  IndexType max_row{i};
742  DataType current_max{std::abs(array_(i_r, i))};
743  // loop with true row index
744  for (IndexType j{i + 1}; j < len; ++j) {
745  // but look up reordered row's value
746  DataType maybe_max{std::abs(array_(row_order_[j], i))};
747 
748  if (maybe_max > current_max) {
749  current_max = std::move(maybe_max);
750  max_row = j;
751  }
752  }
753  // swap if needed. max entry row goes to i-th row.
754  if (max_row != i) {
755  SwapRow(max_row, i);
756  // std::swap(skipmask[maxrow], skipmask[i]);
757  }
758  /* END swap */
759 
760  // forward reduction
761  const DataType a_ii = array_(i_r, i);
762  for (IndexType j{i + 1}; j < len; ++j) {
763  // retrieved reordered j-th row
764  const IndexType& j_r = row_order_[j];
765  // reduction factor for this row
766  const DataType reduction_factor = array_(j_r, i) / a_ii;
767 
768  // matrix reduction
769  // get the beginnings of i and j rows
770  DataType* a_i = &array_(i_r, 0);
771  DataType* a_j = &array_(j_r, 0);
772  // start with setting the i th column zero.
773  a_j[i] = 0.;
774  // this is column entry, so true indices
775  // reduce the rest of the column
776  for (IndexType k{i + 1}; k < len; ++k) {
777  a_j[k] -= a_i[k] * reduction_factor;
778  }
779 
780  // row-vec reduction
781  b[j_r] -= b[i_r] * reduction_factor;
782  }
783  /* END forward reduction */
784  }
785  // back substitution
786  // Loop backwards
787  for (SignedIndexType i{len - 1}; i > static_cast<SignedIndexType>(-1);
788  --i) {
789  // ignore clipped entries
790  // if (skipmask[i] != 0)
791  // continue;
792 
793  const IndexType& i_r = row_order_[i];
794  DataType* a_i = &array_(i_r, 0);
795  DataType sum{};
796  for (IndexType j{i + 1}; j < len; ++j) {
797  sum += a_i[j] * x[j];
798  }
799  x[i] = (b[i_r] - sum) / a_i[i];
800  }
801 
802 #ifndef NDEBUG
803  double norm{};
804  for (IndexType i{}; i < len; ++i) {
805  double row_val{};
806  for (IndexType j{}; j < len; ++j) {
807  row_val += array_copy(i, j) * x[j];
808  }
809  row_val -= b_copy[i];
810  norm += row_val * row_val;
811  }
812 
813  if (std::sqrt(norm) > 1e-4) {
814  splinepy::utils::PrintAndThrowError(
815  "Solving linear system failed, error is : ",
816  norm);
817  }
818 #endif
819  }
820 };

863 
864  /* swap */
865  // go through the rows and compare magnitude and mark it if needed
866  maxrow = i;
867  maxval = std::abs(A[i][i]);
868  for (std::size_t j{i + 1}; j < para_dim; j++) {
869  maybemax = std::abs(A[j][i]);
870  if (maybemax > maxval) {
871  maxval = maybemax;
872  maxrow = j;
873  }
874  }
875  // swap if needed. max entry row goes to i-th row.
876  if (maxrow != i) {
877  // swap A's rows, entries of b, indexmap, and skipmask
878  std::swap(A[maxrow], A[i]);
879  std::swap(b[maxrow], b[i]);
880  std::swap(indexmap[maxrow], indexmap[i]);
881  std::swap(skipmask[maxrow], skipmask[i]);
882  }
883  /* END swap */
884 
885  /* forward reduction */
886  const double Aii = A[i][i];
887  for (std::size_t j{i + 1}; j < para_dim; j++) {
888  const double redfac = A[j][i] / Aii;
889  // matrix reduction
890  for (std::size_t k{i + 1}; k < para_dim; k++) {
891  A[j][k] -= A[i][k] * redfac;
892  }
893  A[j][i] = 0.;
894 
895  // row-vec reduction0
896  b[j] -= b[i] * redfac;
897  }
898  /* END forward reduction */
899  }
900 
901  // back substitution
902  double sum;
903  // Looping backwards, stopping once overflow is reached, which corresponds to
904  // (-1), i.e., such that the last value is 0
905  for (std::size_t i{para_dim - 1}; i != static_cast<std::size_t>(-1); --i) {
906  // ignore clipped entries
907  if (skipmask[i] != 0)
908  continue;
909 
910  sum = 0.;
911  for (std::size_t j{i + 1}; j < para_dim; j++) {
912  sum += A[i][j] * x[j];
913  }
914  x[i] = (b[i] - sum) / A[i][i];
915  }
916 
917  // reorder
918  CopyReorder(x, indexmap);
919  CopyReorder(skipmask, indexmap);
920 }
