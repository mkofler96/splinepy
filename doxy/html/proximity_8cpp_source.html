145 void Proximity::PlantNewKdTree(const int* resolutions, const int n_thread) {
146  const int para_dim = spline_.SplinepyParaDim();
147  const int dim = spline_.SplinepyDim();
148 
149  // get parametric bounds
150  RealArray parametric_bounds(para_dim * 2);
151  spline_.SplinepyParametricBounds(parametric_bounds.data());
152 
153  // create grid_points
154  grid_points_.SetUp(para_dim, parametric_bounds.data(), resolutions);
155 
156  const int n_queries = grid_points_.Size();
157 
158  // reallocate sampled spline
159  sampled_spline_.Reallocate(n_queries * dim);
160 
161  // lambda function to allow n-thread execution
162  auto sample_coordinates = [&](const int begin, const int end, int) {
163  RealArray query(para_dim);
164  double* query_data = query.data();
165 
166  for (int i{begin}; i < end; ++i) {
167  grid_points_.IdToGridPoint(i, query_data);
168  spline_.SplinepyEvaluate(query_data, &sampled_spline_[i * dim]);
169  }
170  };
171 
172  // n-thread execution for sampling
173  splinepy::utils::NThreadExecution(sample_coordinates, n_queries, n_thread);
174 
175  // nanoflann supports concurrent build
176  nanoflann::KDTreeSingleIndexAdaptorParams params{};
177  params.n_thread_build = static_cast<
178  decltype(nanoflann::KDTreeSingleIndexAdaptorParams::n_thread_build)>(
179  (n_thread < 0) ? 0 : n_thread);
180 
181  // plant a new tree
182  cloud_ = std::make_unique<Cloud_>(sampled_spline_.data(),
183  sampled_spline_.size(),
184  dim);
185  kdtree_ = std::make_unique<Tree_>(dim, *cloud_, params);
186 }

239 void Proximity::dJdu(const RealArray& guess,
240  const RealArray& difference,
241  RealArray& djdu,
242  RealArray2D& spline_gradient) const {
243  const int para_dim = guess.size();
244  const int dim = difference.size();
245 
246  IndexArray derivative_query(para_dim);
247  derivative_query.Fill(0);
248 
249  // this is just to view and apply inner product
250  RealArray gradient_row_view;
251  gradient_row_view.SetShape(dim);
252 
253  for (int i{}; i < para_dim; ++i) {
254  // set query - this should be all zero here
255  derivative_query[i] = 1;
256 
257  // set row view
258  gradient_row_view.SetData(&spline_gradient(i, 0));
259 
260  // derivative evaluation
261  spline_.SplinepyDerivative(guess.data(),
262  derivative_query.data(),
263  gradient_row_view.data());
264 
265  // note that there's no negative here, thus not rhs yet
266  djdu[i] = difference.InnerProduct(gradient_row_view);
267 
268  // reset query to zero
269  derivative_query[i] = 0;
270  }
271 }

273 void Proximity::d2Jdu2(const RealArray& guess,
274  const RealArray& difference,
275  const RealArray2D& spline_gradient_AAt,
276  RealArray2D& d2jdu2,
277  RealArray3D& spline_hessian) const {
278  const int para_dim = guess.size();
279  const int dim = difference.size();
280 
281  IndexArray derivative_query(para_dim);
282  derivative_query.Fill(0);
283 
284  // derivative result is a view to the hessian array
285  RealArray derivative;
286  derivative.SetShape(dim);
287 
288  // lambda to compute each element
289  auto compute = [&](const int& i, const int& j) {
290  // adjust derivative query
291  ++derivative_query[i];
292  ++derivative_query[j];
293 
294  // adjust view on hessian array
295  derivative.SetData(&spline_hessian(i, j, 0));
296 
297  // compute
298  spline_.SplinepyDerivative(guess.data(),
299  derivative_query.data(),
300  derivative.data());
301 
302  // fill lhs_ij
303  d2jdu2(i, j) =
304  difference.InnerProduct(derivative) + spline_gradient_AAt(i, j);
305 
306  // reset derivative query
307  --derivative_query[i];
308  --derivative_query[j];
309  };
310 
311  for (int i{}; i < para_dim; ++i) {
312  for (int j{i + 1}; j < para_dim; ++j) {
313  // upper triangle without diagonal
314  compute(i, j);
315  // copy to the lower
316  d2jdu2(j, i) = d2jdu2(i, j);
317  }
318  // diagonal
319  compute(i, i);
320  }
321 }

323 void Proximity::ComputeCostAndDerivatives(SearchData& data, int depth) const {
324  // magic negative will set to the highest depth we use.
325  depth = (depth < 0) ? 2 : depth;
326 
327  if (depth > -1) {
328  data.status.J = J(data.current_guess,
329  data.phys_query,
330  data.difference,
331  data.current_phys);
332  }
333 
334  if (depth > 0) {
335  dJdu(data.current_guess, data.difference, data.djdu, data.spline_gradient);
336  }
337 
338  if (depth > 1) {
339  data.spline_gradient.AAt(data.spline_gradient_AAt);
340  d2Jdu2(data.current_guess,
341  data.difference,
342  data.spline_gradient_AAt,
343  data.d2jdu2,
344  data.spline_hessian);
345  }
346 }

348 void Proximity::ComputeStatus(SearchData& data,
349  const bool recompute_costs) const {
350  if (recompute_costs) {
351  ComputeCostAndDerivatives(data, 1);
352  }
353 
354  // fill status
355  data.status.squared_distance = data.difference.NormL2Squared();
356  data.status.distance = std::sqrt(data.status.squared_distance);
357  data.status.J = 0.5 * data.status.squared_distance;
358  data.status.convergence_norm = data.djdu.NormL2();
359 }

361 void Proximity::FindSearchBound(SearchData& data, const bool tight) const {
362  spline_.SplinepyParametricBounds(data.search_bounds.data());
363 
364  // set search bounds to para bounds and early exit
365  if (!tight) {
366  return;
367  }
368 
369  // you need to be sure that you have sampled your spline fine enough
370  for (int i{}; i < data.para_dim; ++i) {
371  // adjust lower (0) and upper (1) bounds aggressively
372  // but of course, not so aggressive that it is out of bound.
373  data.search_bounds(0, i) =
374  std::max(data.search_bounds(0, i),
375  data.current_guess[i] - grid_points_.step_size_[i]);
376  data.search_bounds(1, i) =
377  std::min(data.search_bounds(1, i),
378  data.current_guess[i] + grid_points_.step_size_[i]);
379  }
380 }

391 void Proximity::Newton(SearchData& data) const {
392  // initialize stop_iteration
393  data.status.stop_iteration = 0;
394 
396  auto& lhs = data.Lhs(data.d2jdu2);
397  auto& rhs = data.djdu;
398 
399  // first round
400  PrepareIterationNewton(data);
401  ComputeStatus(data, false); // compute status based on current values
402 
403  // set convergence goal based on the first round
404  data.options.convergence_goal = ConvergenceGoal(data.status.convergence_norm,
405  data.options.tolerance,
406  data.options.tolerance);
407 
408  int i{};
409  while (i < data.options.max_iter) {
410  if (data.IsConverged()) {
411  break;
412  }
413  // solve systems using gauss elimination with partial pivoting
414  // this will alter lhs in place, but shouldn't reorder rows inplace
415  lhs.ResetRowOrder();
416  lhs.Solve(rhs, data.delta_guess);
417 
418  // x_n+1 = x_n + delta_x
419  UpdateAndClip(data);
420 
421  // Prepare next step and compute status
422  PrepareIterationNewton(data);
423  ComputeStatus(data, false);
424 
425  data.status.stop_iteration = ++i;
426  }
427 }

429 void Proximity::PrepareIterationLevenbergMarquart(SearchData& data) const {
430  // LM requires Jacobian
431  ComputeCostAndDerivatives(data, 1);
432 
433  // prepare rhs
434  data.djdu.FlipSign();
435 
436  // prepare lhs
437  data.spline_gradient.AAt(data.spline_gradient_AAt);
438 
439  // modify lhs based on current value in data.LM
440  double diag_contribution{data.LM.lambda};
441  const bool modified = data.LM.modified_update;
442  if (modified) {
443  diag_contribution += 1.0;
444  }
445 
446  for (int i{}; i < data.para_dim; ++i) {
447  if (modified) {
448  data.spline_gradient_AAt(i, i) *= diag_contribution;
449  } else {
450  data.spline_gradient_AAt(i, i) += diag_contribution;
451  }
452  }
453 }

455 void Proximity::LevenbergMarquart(SearchData& data) const {
456  // initialize stop_iteration
457  data.status.stop_iteration = 0;
458 
459  // for lhs we will use AAt matrix. Modifications may be made to diagonal
460  // entries
461  auto& lhs = data.Lhs(data.spline_gradient_AAt);
462  auto& rhs = data.djdu;
463 
464  // before first round, initialize some options
465  data.LM.lambda = 1.;
466  data.LM.lower_bound = .25;
467  data.LM.upper_bound = .75;
468 
469  PrepareIterationLevenbergMarquart(data);
470  ComputeStatus(data, false);
471 
472  // set initial previous value
473  data.LM.prev_J = data.status.J;
474 
475  // set convergence goal based on the first round
476  data.options.convergence_goal = ConvergenceGoal(data.status.convergence_norm,
477  data.options.tolerance,
478  data.options.tolerance);
479 
480  int i{};
481  data.LM.Setup(data);
482  while (i < data.options.max_iter) {
483  if (data.IsConverged()) {
484  break;
485  }
486  lhs.ResetRowOrder();
487  lhs.Solve(rhs, data.delta_guess);
488 
489  // update the save original_guess before updating
490  data.LM.prev_guess = data.current_guess;
491  UpdateAndClip(data);
492 
493  // get actual delta guess, in case it is clipped
494  if (data.clipped.NonZeros()) {
495  data.delta_guess = data.LM.prev_guess;
496  data.delta_guess.Subtract(data.current_guess);
497  }
498 
499  // compute two different metrics and take bigger one
500  auto& metric = data.LM.Metric(data.spline_gradient, data.delta_guess);
501  const double metric2 = metric.InnerProduct(data.difference);
502 
503  metric.Add(data.difference);
504  const double metric_norm = .5 * metric.NormL2Squared();
505 
506  // compute || F(x + dX) ||^2 then rho
507  ComputeCostAndDerivatives(data, 0); // computes J and everything on the way
508  const double rho =
509  (data.LM.prev_J - data.status.J) / (data.LM.prev_J - metric_norm);
510  const double rho2 = -(data.LM.prev_J - data.status.J) / metric2;
511 
512  const double chosen_rho = std::max(rho, rho2);
513 
514  // If the metric is smaller than the lower bound, the update is
515  // rejected and the penelization is increased
516  if (chosen_rho < data.LM.lower_bound) {
517  // This is @jzwar's practical trick:
518  // Increase lambda if current value is worse than previous
519  if (data.status.J > data.LM.prev_J) {
520  data.LM.lambda *= 2.0;
521  // if it gets too big, exit
522  // if lambda is between 2^13 and 2^14, we exit
523  if (data.LM.lambda > 1.e5) {
524  // status needs re-evaluation at updated location
525  ComputeStatus(data, true);
526  break;
527  }
528 
529  // we continue -> set current_guess back to previous guess
530  data.current_guess = data.LM.prev_guess;
531  PrepareIterationLevenbergMarquart(data);
532  ComputeStatus(data, false);
533  continue;
534  }
535  }
536 
537  // if rho is too big, decrease lambda
538  if (chosen_rho > data.LM.upper_bound) {
539  data.LM.lambda *= 0.5;
540  }
541 
542  // we are here, because lambda is not modified
543  // take this value - current_guess stays the same,
544  data.LM.prev_J = data.status.J;
545 
546  PrepareIterationLevenbergMarquart(data);
547  ComputeStatus(data, false);
548 
549  data.status.stop_iteration = ++i;
550  }
551 }

553 void Proximity::PrepareIterationSlsqp(SearchData& d) const {
554  // precautious step based on
555  // https://github.com/scipy/scipy/issues/11403
556  d.current_guess.Clip(d.lower_bound, d.upper_bound, d.clipped);
557 
558  // mode 0: this is first
559  if (d.SLSQP.mode == 0) {
560  ComputeCostAndDerivatives(d, 1);
561  ComputeStatus(d, false); // strictly not required
562  d.SLSQP.CopyDerivative(d.djdu);
563  d.SLSQP.f = d.status.J;
564  } else if (d.SLSQP.mode == 1) {
565  ComputeCostAndDerivatives(d, 0);
566  d.SLSQP.f = d.status.J;
567  } else if (d.SLSQP.mode == -1) {
568  ComputeCostAndDerivatives(d, 1);
569  d.SLSQP.CopyDerivative(d.djdu);
570  }
571 }

673  MakeInitialGuess(data);
674 
675  // we want to keep the best result
676  ComputeCostAndDerivatives(data, 0); // computes J
677  double best_J = data.status.J;
678  RealArray best_guess(data.initial_guess);
679  auto keep_best_and_did_it_improve = [&best_J, &best_guess, &data]() -> bool {
680  // current guess is better - use the new one
681  if (best_J > data.status.J) {
682  best_guess = data.current_guess;
683  best_J = data.status.J;
684  return true;
685  }
686 
687  // previous guess was better - set back
688  data.current_guess = best_guess;
689  best_J = data.status.J;
690  return false;
691  };
692 
693  // return preps
694  auto fill_hessian_and_metrics =
695  [&data, &distance, &convergence_norm, this](const bool compute) {
696  if (!data.spline_hessian.OwnsData()) {
697  if (compute) {
698  FillHessian(spline_, data.current_guess, data.spline_hessian);
699  } else {
700  splinepy::utils::CopyUpperToLowerTriangle(data.spline_hessian);
701  }
702  }
703 
704  // metrics
705  distance = data.status.distance;
706  convergence_norm = data.status.convergence_norm;
707  };
708 
709  // set bounds
710  FindSearchBound(data, tight_bounds);
711 
712  // we first start with Newton
713  Newton(data);
714 
715  // newton converged - just fill lower triangle of the second_derivative
716  if (data.IsConverged()) {
717  // after newton, we just need to copy upper to lower triangle
718  fill_hessian_and_metrics(false);
719  return;
720  }
721 
722  // newton didn't converge. prepare SLSQP
723  keep_best_and_did_it_improve();
724  data.options.max_iter *= 5;
725  Slsqp(data);
726  // check
727  if (data.IsConverged() || data.SLSQP.mode == 0) {
728  fill_hessian_and_metrics(true);
729  return;
730  }
731 
732  // Newton and Slsqp didn't work. Try LevenbergMarquart
733  keep_best_and_did_it_improve();
734  // set max iteration higher
735  data.options.max_iter *= 5;
736  LevenbergMarquart(data);
737 
738  // if it is improved, recompute all
739  if (keep_best_and_did_it_improve()) {
740  ComputeStatus(data, true); // this should fill grad
741  }
742 
743  // finally, hessian and metrics
744  fill_hessian_and_metrics(true);
745 }

747 void FillHessian(const splinepy::splines::SplinepyBase& spline,
748  const RealArray& at,
749  RealArray3D& hess) {
750  const int para_dim = hess.Shape()[1];
751  const int dim = hess.Shape()[2];
752 
753  IndexArray derivative_query(para_dim);
754  derivative_query.Fill(0);
755 
756  // derivative result is a view to the hessian array
757  RealArray derivative;
758  derivative.SetShape(dim);
759 
760  auto compute = [&](const int& i, const int& j) {
761  // adjust derivative query
762  ++derivative_query[i];
763  ++derivative_query[j];
764 
765  // adjust view on hessian array
766  derivative.SetData(&hess(i, j, 0));
767 
768  // compute
769  spline.SplinepyDerivative(at.data(),
770  derivative_query.data(),
771  derivative.data());
772 
773  // reset derivative query
774  --derivative_query[i];
775  --derivative_query[j];
776  };
777 
778  for (int i{}; i < para_dim; ++i) {
779  for (int j{i + 1}; j < para_dim; ++j) {
780  // upper triangle without diagonal
781  compute(i, j);
782  // copy derivative to lower
783  std::copy_n(derivative.data(), dim, &hess(j, i, 0));
784  }
785  // diagonal
786  compute(i, i);
787  }
788 }
